**STEP 1 — Add `list_history_for_agent` to `services/inapp_messages_store.py`:**

Append a new async function below the existing functions. Match the existing module's style (arg names, return type hints, logging). Concrete shape:

```python
async def list_history_for_agent(
    conn,
    *,
    agent_id: UUID,
    limit: int,
) -> list[dict]:
    """Return terminal-state inapp_messages rows for an agent, oldest first.

    D-03/D-04: status IN ('done', 'failed'); ORDER BY created_at ASC; LIMIT.
    Caller is responsible for clamping `limit` to [1, 1000] before calling
    (handler-level validation per D-04).

    Returns rows shaped:
        [{id, content, status, bot_response, last_error, created_at}, ...]
    """
    rows = await conn.fetch(
        """
        SELECT id, content, status, bot_response, last_error, created_at
        FROM inapp_messages
        WHERE agent_id = $1
          AND status IN ('done', 'failed')
        ORDER BY created_at ASC
        LIMIT $2
        """,
        agent_id,
        limit,
    )
    return [dict(r) for r in rows]
```

```python
from typing import Literal  # if not already imported

class HistoryEvent(BaseModel):
    """One UI-renderable event in the chat history (D-03 + D-04 mapping)."""
    role: Literal["user", "assistant"]
    kind: Literal["message", "error"] = "message"
    content: str
    created_at: datetime
    inapp_message_id: UUID  # for client-side dedup against SSE replay


class HistoryResponse(BaseModel):
    messages: list[HistoryEvent]


_DEFAULT_LIMIT = 200
_MAX_LIMIT = 1000


@router.get("/agents/{agent_id}/messages", status_code=200)
async def get_messages(
    request: Request,
    agent_id: UUID,
    limit: int = _DEFAULT_LIMIT,
):
    # --- Step 1: limit validation (D-04) ---
    if limit < 1:
        return _err(400, ErrorCode.INVALID_REQUEST,
                    "limit must be >= 1", param="limit")
    effective_limit = min(limit, _MAX_LIMIT)

    # --- Step 2: require_user ---
    sess = require_user(request)
    if isinstance(sess, JSONResponse):
        return sess
    user_id: UUID = sess

    # --- Step 3: ownership filter (existing pattern) ---
    pool = request.app.state.db
    async with pool.acquire() as conn:
        ai = await fetch_agent_instance(conn, agent_id=agent_id)
        if ai is None or ai["user_id"] != user_id:
            return _err(404, ErrorCode.AGENT_NOT_FOUND,
                        "agent not found", param="agent_id")

        rows = await list_history_for_agent(
            conn, agent_id=agent_id, limit=effective_limit,
        )

    # --- Step 4: row-to-event mapping (D-03) ---
    events: list[dict] = []
    for r in rows:
        # Always emit the user-side event (every terminal row has a user message).
        events.append({
            "role": "user",
            "kind": "message",
            "content": r["content"],
            "created_at": r["created_at"].isoformat(),
            "inapp_message_id": str(r["id"]),
        })
        # Emit the assistant-side event keyed on status.
        if r["status"] == "done":
            events.append({
                "role": "assistant",
                "kind": "message",
                "content": r["bot_response"] or "",
                "created_at": r["created_at"].isoformat(),
                "inapp_message_id": str(r["id"]),
            })
        elif r["status"] == "failed":
            err_text = r.get("last_error") or "unknown error"
            events.append({
                "role": "assistant",
                "kind": "error",
                "content": f"⚠️ delivery failed: {err_text}",
                "created_at": r["created_at"].isoformat(),
                "inapp_message_id": str(r["id"]),
            })
        # 'pending'/'forwarded' rows are filtered out at the SQL level (D-03).

    return JSONResponse(status_code=200, content={"messages": events})
```
