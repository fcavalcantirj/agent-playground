package session

import (
    "bytes"

    "<module>/internal/recipes"

// ErrTimeout means the agent did not respond within recipe.ChatIO.ResponseTimeout.
var ErrTimeout = errors.New("agent response timeout")

// RunnerExec is the subset of *docker.Runner the bridge needs. Allows mock injection.
// Signature MUST match the runner.go methods exactly — read runner.go first to verify.
type RunnerExec interface {
    Exec(ctx context.Context, containerID string, cmd []string) ([]byte, error)
    ExecWithStdin(ctx context.Context, containerID string, cmd []string, stdin io.Reader) ([]byte, error)
}

type Bridge struct {
    runner RunnerExec
}

func NewBridge(r RunnerExec) *Bridge { return &Bridge{runner: r} }

// SendMessage dispatches based on the recipe's ChatIO mode.
func (b *Bridge) SendMessage(ctx context.Context, containerID string, recipe *recipes.Recipe, text string) (string, error) {
    ctx, cancel := context.WithTimeout(ctx, recipe.ChatIO.ResponseTimeout)
    defer cancel()

    switch recipe.ChatIO.Mode {
    case recipes.ChatIOExec:
        return b.execMode(ctx, containerID, recipe, text)
    case recipes.ChatIOFIFO:
        return b.fifoMode(ctx, containerID, recipe, text)
    default:
        return "", fmt.Errorf("unknown chat io mode: %s", recipe.ChatIO.Mode)
    }
}

func (b *Bridge) execMode(ctx context.Context, containerID string, recipe *recipes.Recipe, text string) (string, error) {
    cmd := append(slices.Clone(recipe.ChatIO.ExecCmd), text)
    out, err := b.runner.Exec(ctx, containerID, cmd)
    if err != nil {
        if errors.Is(err, context.DeadlineExceeded) {
            return "", ErrTimeout
        }
        return "", fmt.Errorf("exec: %w", err)
    }
    return stripANSI(strings.TrimSpace(string(out))), nil
}

func (b *Bridge) fifoMode(ctx context.Context, containerID string, recipe *recipes.Recipe, text string) (string, error) {
    payload := bytes.NewReader([]byte(text + "\n"))
    if _, err := b.runner.ExecWithStdin(ctx, containerID,
        []string{"sh", "-c", "cat >> /run/ap/chat.in"}, payload); err != nil {
        return "", fmt.Errorf("fifo write: %w", err)
    }
    for {
        if err := ctx.Err(); err != nil {
            return "", ErrTimeout
        }
        out, err := b.runner.Exec(ctx, containerID,
            []string{"timeout", "5", "head", "-n", "1", "/run/ap/chat.out"})
        if err == nil && len(bytes.TrimSpace(out)) > 0 {
            return strings.TrimSpace(string(out)), nil
        }
        select {
        case <-ctx.Done():
            return "", ErrTimeout
        case <-time.After(100 * time.Millisecond):
        }
    }
}

func stripANSI(s string) string {
    var out strings.Builder
    inEscape := false
    for _, r := range s {
        if r == '\x1b' { inEscape = true; continue }
        if inEscape {
            if (r >= 'a' && r <= 'z') || (r >= 'A' && r <= 'Z') { inEscape = false }
            continue
        }
        out.WriteRune(r)
    }
    return out.String()
}

| Boundary | Description |
|----------|-------------|
| HTTP client → /api/sessions | User input crosses the auth middleware into the session handler; recipe name and message text are user-controlled. |
| Handler → runner.Run | Handler-built RunOptions cross from Go process to Docker daemon; sandbox knobs must be set correctly. |

| T-02-02 | E — Elevation of privilege | Container missing sandbox defaults | mitigate | Handler always builds opts via DefaultSandbox() then layers recipe overrides; CapDrop=ALL, NoNewPrivs=true, ReadOnlyRootfs=true are unconditional; recipe ResourceOverrides only mutate Memory/CPUs/PidsLimit, never the security knobs. |

| T-02-04b | T — Tampering / injection | Chat text via exec argv to hermes | mitigate | ExecCmd is appended via `slices.Clone(execCmd)` then `append(_, text)` — text becomes a single argv element that the SDK passes verbatim. No shell, no interpolation. |
