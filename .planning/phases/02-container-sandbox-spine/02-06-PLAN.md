```bash
#!/usr/bin/env bash
# scripts/smoke-e2e.sh — Phase 2 hypothesis-proof smoke test.
#
# Exercises the API-driven agent start path end-to-end:
#   1. Verify dev compose stack is up (postgres + redis)
#   2. Verify ap-base + recipe images are present (build if missing)
#   3. Start the Go API in the background
#   4. Obtain a dev session cookie via /api/dev/login
#   5. POST /api/sessions with the requested recipe → assert 201 + capture id
#   6. POST /api/sessions/<id>/message with a real prompt → assert 200 + non-empty text
#   7. DELETE /api/sessions/<id> → assert 200
#   8. Assert ZERO dangling playground-* containers
#   9. Tear down API
#
# CONTEXT D-32, D-33, D-34. The curl output is the demo.

set -euo pipefail

AGENT="${1:-}"
if [[ -z "$AGENT" ]]; then
    echo "usage: $0 <picoclaw|hermes>"
    exit 2
fi
if [[ "$AGENT" != "picoclaw" && "$AGENT" != "hermes" ]]; then
    echo "error: unknown agent '$AGENT'"
    exit 2
fi

if [[ -z "${AP_DEV_BYOK_KEY:-}" ]]; then
    echo "SKIPPED: AP_DEV_BYOK_KEY not set (set to a real Anthropic key to run the live smoke test)"
    exit 0
fi

REPO_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
cd "$REPO_ROOT"

API_PORT="${API_PORT:-8080}"
API_BASE="http://localhost:${API_PORT}"
COOKIE_JAR="$(mktemp -t ap-smoke-cookie.XXXXXX)"
API_LOG="$(mktemp -t ap-smoke-api.XXXXXX.log)"
API_PID=""

# Always cleanup on exit (success or failure).
cleanup() {
    local rc=$?
    if [[ -n "$API_PID" ]]; then
        kill "$API_PID" 2>/dev/null || true
        wait "$API_PID" 2>/dev/null || true
    fi
    # Best-effort: remove any leaked containers from a botched test
    docker ps -a --filter name=playground- --format '{{.ID}}' | xargs -r docker rm -f >/dev/null 2>&1 || true
    rm -f "$COOKIE_JAR"
    if [[ $rc -ne 0 ]]; then
        echo
        echo "=== API log tail ==="
        tail -100 "$API_LOG" || true
    fi
    exit $rc
}
trap cleanup EXIT INT TERM

log() { echo "[smoke:$AGENT] $*"; }
fail() { echo "[smoke:$AGENT] FAIL: $*" >&2; exit 1; }

# --- Step 1: dev stack ---
log "verifying dev compose stack"
if ! docker compose -f docker-compose.dev.yml ps --services --filter status=running | grep -q postgres; then
    log "starting docker-compose dev stack"
    docker compose -f docker-compose.dev.yml up -d postgres redis
    sleep 3
fi

# --- Step 2: images ---
log "verifying ap-base + ap-${AGENT} images"
if ! docker images "ap-base:v0.1.0" --format '{{.Repository}}' | grep -q '^ap-base$'; then
    log "building ap-base"
    make build-ap-base
fi
if ! docker images "ap-${AGENT}:v0.1.0-*" --format '{{.Repository}}' | grep -q "^ap-${AGENT}$"; then
    log "building ap-${AGENT}"
    make "build-${AGENT}"
fi

# --- Step 3: kill any stale process from a previous run, then start API ---
log "killing any stale API on :${API_PORT}"
lsof -ti tcp:"${API_PORT}" 2>/dev/null | xargs -r kill -9 || true

log "starting Go API"
cd api
AP_DEV_MODE=true \
AP_SESSION_SECRET=test-secret-that-is-at-least-32-characters-long \
DATABASE_URL="postgres://temporal:temporal@localhost:5432/agent_playground?sslmode=disable" \
REDIS_URL=redis://localhost:6379 \
TEMPORAL_HOST= \
API_PORT="${API_PORT}" \
AP_DEV_BYOK_KEY="${AP_DEV_BYOK_KEY}" \
go run ./cmd/server/ > "$API_LOG" 2>&1 &
API_PID=$!
cd "$REPO_ROOT"

# Wait for /healthz
for i in {1..30}; do
    if curl -fsS "${API_BASE}/healthz" >/dev/null 2>&1; then
        log "API healthy after ${i}s"
        break
    fi
    sleep 1
done
if ! curl -fsS "${API_BASE}/healthz" >/dev/null; then
    fail "API never became healthy on ${API_BASE}/healthz"
fi

# --- Step 4: dev login → cookie ---
log "obtaining dev session cookie"
curl -fsS -c "$COOKIE_JAR" -X POST -H 'Content-Type: application/json' \
    -d '{"email":"smoke@example.com"}' \
    "${API_BASE}/api/dev/login" > /dev/null \
    || fail "dev login failed"
grep -q ap_session "$COOKIE_JAR" || fail "no ap_session cookie set"

# --- Step 5: POST /api/sessions ---
log "creating session for recipe=${AGENT}"
CREATE_RESP="$(curl -fsS -b "$COOKIE_JAR" -X POST \
    -H 'Content-Type: application/json' \
    -d "{\"recipe\":\"${AGENT}\",\"model_provider\":\"anthropic\",\"model_id\":\"claude-sonnet-4-6\"}" \
    "${API_BASE}/api/sessions")"
echo "  → $CREATE_RESP"
SESSION_ID="$(echo "$CREATE_RESP" | grep -o '"id":"[^"]*"' | head -1 | sed 's/.*:"//;s/"//')"
[[ -n "$SESSION_ID" ]] || fail "no session id in create response"
echo "$CREATE_RESP" | grep -q '"status":"running"' || fail "session not in running state after create"

# --- Step 6: POST /api/sessions/<id>/message — the hypothesis proof ---
log "sending real prompt to ${AGENT} (this is the demo)"
MESSAGE_RESP="$(curl -fsS -b "$COOKIE_JAR" -X POST \
    -H 'Content-Type: application/json' \
    -d '{"text":"In exactly five words, say hello."}' \
    "${API_BASE}/api/sessions/${SESSION_ID}/message")" \
    || fail "message POST failed (see API log)"
echo
echo "================ AGENT RESPONSE ================"
echo "$MESSAGE_RESP"
echo "================================================"
echo

# Assert non-empty text field
RESPONSE_TEXT="$(echo "$MESSAGE_RESP" | grep -o '"text":"[^"]*"' | head -1 | sed 's/.*:"//;s/"$//')"
[[ -n "$RESPONSE_TEXT" ]] || fail "empty text in message response"
echo "$RESPONSE_TEXT" | grep -qi 'error' && fail "response text looks like an error: $RESPONSE_TEXT"
log "got non-empty response (${#RESPONSE_TEXT} chars)"

# --- Step 7: DELETE /api/sessions/<id> ---
log "deleting session ${SESSION_ID}"
curl -fsS -b "$COOKIE_JAR" -X DELETE "${API_BASE}/api/sessions/${SESSION_ID}" \
    | grep -q '"status":"stopped"' \
    || fail "delete did not return stopped status"

# --- Step 8: ZERO dangling containers ---
sleep 2
DANGLING="$(docker ps -a --filter name=playground- --format '{{.Names}}')"
if [[ -n "$DANGLING" ]]; then
    fail "dangling playground containers after delete: $DANGLING"
fi
log "zero dangling playground-* containers — clean teardown verified"

# --- Step 9: secret cleanup ---
if [[ -d "/tmp/ap/secrets/${SESSION_ID}" ]]; then
    fail "secret dir not cleaned: /tmp/ap/secrets/${SESSION_ID}"
fi
log "secret dir cleanup verified"

log "PASS: ${AGENT} hypothesis proof"
```

package session_test

import (

    "errors"
    "testing"

    "github.com/agent-playground/api/internal/session"

    embeddedpostgres "github.com/fergusstrange/embedded-postgres"

    "github.com/stretchr/testify/require"

// (Skeleton — actual implementation must match Phase 1's existing
// embedded-postgres test helper. If Phase 1 has a testutil package for
// this, USE IT instead of duplicating boot logic.)

func setupDB(t *testing.T) (*pgxpool.Pool, func()) {
    t.Helper()
    pg := embeddedpostgres.NewDatabase(embeddedpostgres.DefaultConfig().Port(54329))
    require.NoError(t, pg.Start())
    pool, err := pgxpool.New(context.Background(), "postgres://postgres:postgres@localhost:54329/postgres?sslmode=disable")
    require.NoError(t, err)
    require.NoError(t, migrate.Run(context.Background(), pool))
    return pool, func() { pool.Close(); _ = pg.Stop() }
}

func mustCreateUser(t *testing.T, pool *pgxpool.Pool) uuid.UUID {
    t.Helper()
    id := uuid.New()
    _, err := pool.Exec(context.Background(),
        `INSERT INTO users (id, provider, provider_sub, email) VALUES ($1, 'dev', $2, $3)`,
        id, id.String(), id.String()+"@example.com")
    require.NoError(t, err)
    return id
}

func TestSessionLifecycle_OneActivePerUser(t *testing.T) {
    pool, teardown := setupDB(t)
    defer teardown()
    store := session.NewStore(pool)

    user := mustCreateUser(t, pool)
    s1, err := store.Create(context.Background(), user, "picoclaw", "anthropic", "claude-sonnet-4-6")
    require.NoError(t, err)
    require.NotNil(t, s1)

    _, err = store.Create(context.Background(), user, "picoclaw", "anthropic", "claude-sonnet-4-6")
    require.ErrorIs(t, err, session.ErrConflictActive)
}

func TestSessionLifecycle_AfterStopAllowsNew(t *testing.T) {
    pool, teardown := setupDB(t)
    defer teardown()
    store := session.NewStore(pool)

    user := mustCreateUser(t, pool)
    s1, err := store.Create(context.Background(), user, "picoclaw", "anthropic", "claude-sonnet-4-6")
    require.NoError(t, err)

    require.NoError(t, store.UpdateStatus(context.Background(), s1.ID, session.StatusStopped))

    s2, err := store.Create(context.Background(), user, "hermes", "anthropic", "claude-sonnet-4-6")
    require.NoError(t, err)
    require.NotEqual(t, s1.ID, s2.ID)
}

func TestSessionLifecycle_DistinctUsersConcurrent(t *testing.T) {
    pool, teardown := setupDB(t)
    defer teardown()
    store := session.NewStore(pool)

    u1 := mustCreateUser(t, pool)
    u2 := mustCreateUser(t, pool)

    _, err := store.Create(context.Background(), u1, "picoclaw", "anthropic", "claude-sonnet-4-6")
    require.NoError(t, err)
    _, err = store.Create(context.Background(), u2, "picoclaw", "anthropic", "claude-sonnet-4-6")
    require.NoError(t, err, "distinct users must each be allowed one active session")

    if errors.Is(err, session.ErrConflictActive) {
        t.Fatal("partial unique index incorrectly fired across users")
    }
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Smoke test → real Anthropic API | The test sends a real prompt to api.anthropic.com via the agent inside the container. The key crosses the entire injection chain: shell env → server env → host file → bind mount → entrypoint reader → agent process env. |
| AP_DEV_BYOK_KEY in developer shell history | Test depends on the env var; developers may accidentally `history | grep` and leak. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | I — Information disclosure | Smoke test logs the BYOK key | mitigate | Script never echoes `$AP_DEV_BYOK_KEY` directly; it only checks `[[ -z … ]]` and passes the value via env to the API child process. The API log capture (`$API_LOG`) is deleted on cleanup. |
| T-02-05 | D — DoS | Botched test leaves dangling containers | mitigate | `trap cleanup EXIT` always runs; cleanup force-removes any `playground-*` containers found. Final assertion verifies zero dangling. |
| T-02-12 | T — Tampering | Smoke test could be invoked with a hostile agent name (e.g. `;rm -rf /`) | mitigate | Script validates `$AGENT` against a fixed allowlist (`picoclaw|hermes`) before any docker invocation. |
| T-02-13 | I — Information disclosure | API log tail printed on failure could include the secret | accept | API logs from Plan 04 must NEVER include the BYOK key value (verified via grep gate in Plan 04 acceptance criteria); the tail is shown only on failure to aid debugging; risk is bounded if Plan 04's no-log invariant holds. |

</threat_model>
