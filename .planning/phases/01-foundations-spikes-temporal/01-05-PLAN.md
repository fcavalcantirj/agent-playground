    4. Create `api/internal/temporal/worker.go`:
       ```go
       package temporal

       import (
           "fmt"
           "github.com/rs/zerolog"
           "go.temporal.io/sdk/client"
           "go.temporal.io/sdk/worker"
       )

       const (
           SessionQueue        = "session"
           BillingQueue        = "billing"
           ReconciliationQueue = "reconciliation"
       )

       // Workers holds the Temporal client and all workers.
       type Workers struct {
           Client  client.Client
           workers []worker.Worker
           logger  zerolog.Logger
       }

       // NewWorkers creates a Temporal client and three workers.
       // Each worker polls its own task queue and has workflows/activities registered.
       func NewWorkers(temporalHost, namespace string, logger zerolog.Logger) (*Workers, error) {
           c, err := client.Dial(client.Options{
               HostPort:  temporalHost,
               Namespace: namespace,
               Logger:    newZerologAdapter(logger), // adapter to bridge zerolog -> Temporal logger
           })
           if err != nil {
               return nil, fmt.Errorf("temporal client dial: %w", err)
           }

           w := &Workers{Client: c, logger: logger}

           // Session worker
           sessionWorker := worker.New(c, SessionQueue, worker.Options{})
           sessionWorker.RegisterWorkflow(PingPong)
           sessionWorker.RegisterWorkflow(SessionSpawn)
           sessionWorker.RegisterWorkflow(SessionDestroy)
           sessionWorker.RegisterWorkflow(RecipeInstall)
           sessionWorker.RegisterActivity(PingActivity)
           sessionWorker.RegisterActivity(SpawnContainerActivity)
           sessionWorker.RegisterActivity(DestroyContainerActivity)
           sessionWorker.RegisterActivity(InstallRecipeActivity)
           w.workers = append(w.workers, sessionWorker)

           // Billing worker
           billingWorker := worker.New(c, BillingQueue, worker.Options{})
           billingWorker.RegisterWorkflow(ReconcileBilling)
           w.workers = append(w.workers, billingWorker)

           // Reconciliation worker
           reconWorker := worker.New(c, ReconciliationQueue, worker.Options{})
           reconWorker.RegisterWorkflow(ReconcileContainers)
           w.workers = append(w.workers, reconWorker)

           return w, nil
       }

       // Start starts all workers. Non-blocking.
       func (w *Workers) Start() error {
           for _, wr := range w.workers {
               if err := wr.Start(); err != nil {
                   return fmt.Errorf("start worker: %w", err)
               }
           }
           w.logger.Info().Msg("all temporal workers started")
           return nil
       }

       // Stop stops all workers gracefully.
       func (w *Workers) Stop() {
           for _, wr := range w.workers {
               wr.Stop()
           }
           w.Client.Close()
           w.logger.Info().Msg("all temporal workers stopped")
       }
       ```

       type zerologAdapter struct {

       func newZerologAdapter(l zerolog.Logger) *zerologAdapter {
           return &zerologAdapter{logger: l.With().Str("component", "temporal").Logger()}

       func (z *zerologAdapter) Debug(msg string, keyvals ...interface{}) {

       func (z *zerologAdapter) Info(msg string, keyvals ...interface{}) {

       func (z *zerologAdapter) Warn(msg string, keyvals ...interface{}) {

       }
       func (z *zerologAdapter) Error(msg string, keyvals ...interface{}) {

    1. Add `WithWorkers` functional option to `api/internal/server/server.go`:
       ```go
       // WithWorkers configures the server with Temporal workers.
       // The workers are stored on the Server struct for lifecycle management.
       // Pass nil to explicitly disable Temporal (same as omitting the option).
       func WithWorkers(w *apitemporal.Workers) Option {
           return func(s *Server) {
               s.Workers = w
           }
       }
       ```
       This requires importing `api/internal/temporal` with alias `apitemporal` in server.go. Alternatively, if you want to avoid a circular import, use the `interface{ Start() error; Stop() }` type that Plan 01-01 defined for the Workers field, and let the caller pass `*apitemporal.Workers` which satisfies it. Choose the approach that avoids import cycles.

       **Preferred approach to avoid circular imports:** The Server.Workers field is typed as `interface{ Start() error; Stop() }`. The WithWorkers option accepts that same interface type:
       ```go
       // WorkerManager is satisfied by *temporal.Workers (and test mocks).
       type WorkerManager interface {
           Start() error
           Stop()
       }

       func WithWorkers(w WorkerManager) Option {
           return func(s *Server) {
               if w != nil {
                   s.Workers = w
               }
           }
       }
       ```
