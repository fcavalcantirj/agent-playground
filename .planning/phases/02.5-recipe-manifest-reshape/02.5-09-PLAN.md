Wire the entire 02.5 substrate into the HTTP handler layer:

1. **New endpoints:** `GET /api/recipes` (list with filter params) and `GET /api/recipes/:id` (detail). Both return public-metadata views — no lifecycle/install/auth/isolation leaks.
2. **Extended endpoint:** `POST /api/sessions` now takes a `provider` field (required when recipe supports multiple providers). Validates against recipe.providers[] and model against recipe.models[].
3. **Sessions handler swap:** the Phase 2 sessions handler currently dispatches to hardcoded `recipes.Picoclaw` / `recipes.Hermes` structs. This plan replaces that with a call path that: (a) fetches the recipe via `server.RecipeLoader.Get(id)`, (b) runs `session.Materialize` to resolve secrets + render templates, (c) builds `docker.RunOptions` from `recipe.Isolation` + `recipe.Runtime.Image` + `MaterializedRecipe.Env`, (d) calls `Runner.RunWithLifecycle`, (e) stores `LifecycleSession.ContainerID` in the sessions table with the chosen provider.
4. **Error surface:** centralize the new error codes in `handler/errors.go` and wire them to the JSON envelope. Phase 1's existing `{"error": {"code": "...", "message": "..."}}` shape is preserved.
5. **Migration 0003:** adds a `provider text NOT NULL DEFAULT 'anthropic'` column to `sessions`. Idempotent (uses `ADD COLUMN IF NOT EXISTS`).

```go
package handler

import (
    "net/http"
    "strings"

    "github.com/agentplayground/api/internal/recipes"
    "github.com/labstack/echo/v4"
)

// RecipeLoaderIface is the narrow interface the handler needs.
// Implemented by *recipes.Loader.
type RecipeLoaderIface interface {
    All() []*recipes.Recipe
    Get(id string) (*recipes.Recipe, bool)
}

type RecipePublicView struct {
    ID          string                  `json:"id"`
    Name        string                  `json:"name"`
    Description string                  `json:"description,omitempty"`
    License     string                  `json:"license,omitempty"`
    Category    string                  `json:"category,omitempty"`
    Runtime     RuntimePublicView       `json:"runtime"`
    Frontend    *recipes.RecipeFrontend `json:"frontend,omitempty"`
    PolicyFlags []string                `json:"policy_flags,omitempty"`
    TierBadge   string                  `json:"tier_badge,omitempty"`
    Providers   []ProviderPublicView    `json:"providers"`
    Models      []ModelPublicView       `json:"models"`
}

type RuntimePublicView struct {
    Family string `json:"family"`
}

type ProviderPublicView struct {
    ID string `json:"id"`
}

type ModelPublicView struct {
    ID       string `json:"id"`
    Provider string `json:"provider"`
}

func ToPublicView(r *recipes.Recipe) RecipePublicView {
    providers := make([]ProviderPublicView, 0, len(r.Providers))
    for _, p := range r.Providers {
        providers = append(providers, ProviderPublicView{ID: p.ID})
    }
    models := make([]ModelPublicView, 0, len(r.Models))
    for _, m := range r.Models {
        models = append(models, ModelPublicView{ID: m.ID, Provider: m.Provider})
    }
    var fe *recipes.RecipeFrontend
    if r.Frontend != (recipes.RecipeFrontend{}) {
        feCopy := r.Frontend
        fe = &feCopy
    }
    return RecipePublicView{
        ID:          r.ID,
        Name:        r.Name,
        Description: r.Description,
        License:     r.License,
        Category:    r.Category,
        Runtime:     RuntimePublicView{Family: r.Runtime.Family},
        Frontend:    fe,
        PolicyFlags: r.PolicyFlags,
        TierBadge:   r.TierBadge,
        Providers:   providers,
        Models:      models,
    }
}

func RegisterRecipesRoutes(e *echo.Group, loader RecipeLoaderIface) {
    e.GET("/recipes", func(c echo.Context) error {
        all := loader.All()
        filtered := filterRecipes(all, c)
        out := make([]RecipePublicView, 0, len(filtered))
        for _, r := range filtered {
            out = append(out, ToPublicView(r))
        }
        return c.JSON(http.StatusOK, out)
    })
    e.GET("/recipes/:id", func(c echo.Context) error {
        id := c.Param("id")
        r, ok := loader.Get(id)
        if !ok {
            return WriteError(c, http.StatusNotFound, ErrCodeRecipeNotFound, "recipe "+id+" not found")
        }
        return c.JSON(http.StatusOK, ToPublicView(r))
    })
}

func filterRecipes(all []*recipes.Recipe, c echo.Context) []*recipes.Recipe {
    family := c.QueryParam("family")
    tier := c.QueryParam("tier")
    license := c.QueryParam("license")
    provider := c.QueryParam("provider")

    out := make([]*recipes.Recipe, 0, len(all))
    for _, r := range all {
        if family != "" && r.Runtime.Family != family {
            continue
        }
        if tier != "" && r.Isolation.Tier != tier {
            continue
        }
        if license != "" && !strings.EqualFold(r.License, license) {
            continue
        }
        if provider != "" {
            found := false
            for _, p := range r.Providers {
                if p.ID == provider {
                    found = true
                    break
                }
            }
            if !found {
                continue
            }
        }
        out = append(out, r)
    }
    return out
}
```

```go
func (h *SessionsHandler) CreateSession(c echo.Context) error {
    var req CreateSessionReq
    if err := c.Bind(&req); err != nil {
        return WriteError(c, http.StatusBadRequest, ErrCodeInvalidRequest, err.Error())
    }
    // Fold Phase 2 aliases
    if req.Provider == "" && req.ModelProvider != "" {
        req.Provider = req.ModelProvider
    }
    if req.Model == "" && req.ModelID != "" {
        req.Model = req.ModelID
    }
    if req.Recipe == "" {
        return WriteError(c, http.StatusBadRequest, ErrCodeInvalidRequest, "recipe is required")
    }
    recipe, ok := h.loader.Get(req.Recipe)
    if !ok {
        return WriteError(c, http.StatusNotFound, ErrCodeRecipeNotFound, "recipe "+req.Recipe+" not found")
    }
    // Default provider if singular
    if req.Provider == "" {
        if len(recipe.Providers) == 1 {
            req.Provider = recipe.Providers[0].ID
        } else {
            return WriteError(c, http.StatusBadRequest, ErrCodeInvalidRequest, "provider is required (recipe has multiple)")
        }
    }
    // Validate provider declared
    providerOK := false
    for _, p := range recipe.Providers {
        if p.ID == req.Provider { providerOK = true; break }
    }
    if !providerOK {
        return WriteError(c, http.StatusBadRequest, ErrCodeProviderNotSupported, "provider "+req.Provider+" not declared by recipe")
    }
    // Default model if not specified: first model matching the provider (D-48 convention)
    if req.Model == "" {
        for _, m := range recipe.Models {
            if m.Provider == req.Provider { req.Model = m.ID; break }
        }
    }
    // Validate model declared AND matches provider
    modelOK := false
    for _, m := range recipe.Models {
        if m.ID == req.Model && m.Provider == req.Provider { modelOK = true; break }
    }
    if !modelOK {
        return WriteError(c, http.StatusBadRequest, ErrCodeModelNotSupported, "model "+req.Model+" not supported by "+req.Provider)
    }
    // Materialize
    mat, err := session.Materialize(c.Request().Context(), recipe, req.Provider, req.Model, h.secrets, h.templates)
    if err != nil {
        switch {
        case errors.Is(err, session.ErrSecretMissing):
            return WriteError(c, http.StatusBadRequest, ErrCodeSecretMissing, err.Error())
        case strings.Contains(err.Error(), "template") || errors.Is(err, recipes.ErrTemplatePath) || errors.Is(err, recipes.ErrTemplateTimeout) || errors.Is(err, recipes.ErrTemplateSize):
            return WriteError(c, http.StatusInternalServerError, ErrCodeTemplateRenderFailed, err.Error())
        default:
            return WriteError(c, http.StatusInternalServerError, ErrCodeInternal, err.Error())
        }
    }
    // Build RunOptions + write materialized files to per-session tmpfs + bind-mount
    runOpts, cleanup, err := h.buildRunOptions(c.Request().Context(), userID, sessionID, recipe, mat)
    if err != nil {
        return WriteError(c, http.StatusInternalServerError, ErrCodeInternal, err.Error())
    }
    defer func() {
        if err != nil && cleanup != nil { cleanup() }
    }()
    lcSession, err := h.runner.RunWithLifecycle(c.Request().Context(), recipe, runOpts)
    if err != nil {
        return WriteError(c, http.StatusInternalServerError, ErrCodeLifecycleHookFailed, err.Error())
    }
    // INSERT session row
    // ... (Phase 2's existing DB insert, with model_provider now = req.Provider)
    return c.JSON(http.StatusCreated, CreateSessionResp{ /* ... */ })
}
```
