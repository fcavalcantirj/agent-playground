```go
package recipes

import (
    "fmt"
    "time"
)

// Hook is the normalized representation of any lifecycle hook.
// Each outer element is one parallel group. Each inner element is the argv
// of a single `docker exec` call. To run multiple commands sequentially,
// wrap them in sh -c: ["sh", "-c", "cmd1 && cmd2"].
//
// v0.1 recipe authoring contract (D-27):
//   hook: "echo hi"                   → one group, one argv, sh -c wrapped
//   hook: ["bin","--flag","arg"]      → one group, one argv, literal exec
//   hook: [["a"],["b"]]               → TWO PARALLEL groups (errgroup dispatches)
//   hook: [["a","arg"],["b","arg"]]   → TWO PARALLEL groups, each multi-arg
//
// Dev Containers object-syntax parallel groups ({"label": "cmd"}) are
// rejected at schema-validation time (Plan 01's hook $def has not:{type:object}).
type Hook [][]string

const DefaultHookTimeout = 10 * time.Minute

// NormalizeHook converts the untyped lifecycle field into a Hook.
func NormalizeHook(v any) (Hook, error) {
    switch x := v.(type) {
    case nil:
        return nil, nil
    case string:
        if x == "" {
            return nil, nil
        }
        return Hook{{"sh", "-c", x}}, nil
    case []any:
        if len(x) == 0 {
            return nil, nil
        }
        // Peek at first element to decide: all-strings → one argv, all-arrays → parallel groups.
        switch x[0].(type) {
        case string:
            argv := make([]string, 0, len(x))
            for _, item := range x {
                s, ok := item.(string)
                if !ok {
                    return nil, fmt.Errorf("hook: mixed type in []string form: got %T", item)
                }
                argv = append(argv, s)
            }
            return Hook{argv}, nil
        case []any:
            out := make(Hook, 0, len(x))
            for _, item := range x {
                inner, ok := item.([]any)
                if !ok {
                    return nil, fmt.Errorf("hook: mixed-type array-of-arrays: got %T", item)
                }
                argv := make([]string, 0, len(inner))
                for _, s := range inner {
                    str, ok := s.(string)
                    if !ok {
                        return nil, fmt.Errorf("hook: non-string in inner argv: got %T", s)
                    }
                    argv = append(argv, str)
                }
                out = append(out, argv)
            }
            return out, nil
        default:
            return nil, fmt.Errorf("hook: unsupported first element type %T", x[0])
        }
    default:
        return nil, fmt.Errorf("hook: unsupported root type %T", v)
    }
}

// HookTimeout returns the effective timeout for the named hook.
func HookTimeout(r *Recipe, name string) time.Duration {
    if r == nil {
        return DefaultHookTimeout
    }
    var sec int
    switch name {
    case "initializeCommand":
        sec = r.Lifecycle.InitializeTimeoutSec
    case "onCreateCommand":
        sec = r.Lifecycle.OnCreateTimeoutSec
    case "updateContentCommand":
        sec = r.Lifecycle.UpdateContentTimeoutSec
    case "postCreateCommand":
        sec = r.Lifecycle.PostCreateTimeoutSec
    case "postStartCommand":
        sec = r.Lifecycle.PostStartTimeoutSec
    case "postAttachCommand":
        sec = r.Lifecycle.PostAttachTimeoutSec
    }
    if sec <= 0 {
        return DefaultHookTimeout
    }
    return time.Duration(sec) * time.Second
}
```

2. Create `api/pkg/docker/runner_lifecycle.go` implementing Pattern 5 from RESEARCH. Key structure:

```go
package docker

import (
    "context"
    "errors"
    "fmt"
    "os/exec"
    "time"

    "github.com/agentplayground/api/internal/recipes"
    "golang.org/x/sync/errgroup"
)

type LifecycleSession struct {
    ContainerID string
    Recipe      *recipes.Recipe
    ReadyCh     chan struct{}
}

// execFunc is the minimal exec contract — production passes Runner.Exec,
// tests pass a fake.
type execFunc func(ctx context.Context, containerID string, cmd []string) ([]byte, error)

func (r *Runner) RunWithLifecycle(
    ctx context.Context,
    recipe *recipes.Recipe,
    opts RunOptions,
) (*LifecycleSession, error) {
    // 1. initializeCommand — host-side.
    if err := r.runHostInitialize(ctx, recipe); err != nil {
        return nil, fmt.Errorf("initializeCommand: %w", err)
    }

    // 2. Create container via existing Phase 2 runner.
    cid, err := r.Run(ctx, opts)
    if err != nil {
        return nil, fmt.Errorf("container create: %w", err)
    }

    session := &LifecycleSession{
        ContainerID: cid,
        Recipe:      recipe,
        ReadyCh:     make(chan struct{}),
    }

    waitFor := recipe.Lifecycle.WaitFor
    if waitFor == "" {
        waitFor = "postCreateCommand"
    }

    // 3-7. In-container hooks.
    alreadyCreated := opts.Labels["ap.already-created"] == "true"
    hooks := []struct {
        name string
        skip bool
    }{
        {"onCreateCommand", alreadyCreated},
        {"updateContentCommand", false},
        {"postCreateCommand", false},
        {"postStartCommand", false},
        {"postAttachCommand", false},
    }

    for _, h := range hooks {
        if !h.skip {
            if err := r.runHookByName(ctx, cid, recipe, h.name, r.Exec); err != nil {
                if terr := r.teardown(ctx, cid); terr != nil {
                    return nil, errors.Join(fmt.Errorf("%s: %w", h.name, err), fmt.Errorf("teardown: %w", terr))
                }
                return nil, fmt.Errorf("%s: %w", h.name, err)
            }
        }
        if h.name == waitFor {
            close(session.ReadyCh)
        }
    }

    return session, nil
}

func (r *Runner) runHookByName(ctx context.Context, cid string, recipe *recipes.Recipe, name string, exec execFunc) error {
    var raw any
    switch name {
    case "onCreateCommand":
        raw = recipe.Lifecycle.OnCreateCommand
    case "updateContentCommand":
        raw = recipe.Lifecycle.UpdateContentCommand
    case "postCreateCommand":
        raw = recipe.Lifecycle.PostCreateCommand
    case "postStartCommand":
        raw = recipe.Lifecycle.PostStartCommand
    case "postAttachCommand":
        raw = recipe.Lifecycle.PostAttachCommand
    default:
        return fmt.Errorf("unknown hook %q", name)
    }
    hook, err := recipes.NormalizeHook(raw)
    if err != nil {
        return fmt.Errorf("normalize: %w", err)
    }
    if len(hook) == 0 {
        return nil
    }
    timeout := recipes.HookTimeout(recipe, name)
    hctx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()
    g, gctx := errgroup.WithContext(hctx)
    for _, argv := range hook {
        argv := argv
        g.Go(func() error {
            out, err := exec(gctx, cid, argv)
            if err != nil {
                return fmt.Errorf("exec %v: %w (output=%q)", argv, err, string(out))
            }
            return nil
        })
    }
    return g.Wait()
}

func (r *Runner) runHostInitialize(ctx context.Context, recipe *recipes.Recipe) error {
    hook, err := recipes.NormalizeHook(recipe.Lifecycle.InitializeCommand)
    if err != nil {
        return err
    }
    if len(hook) == 0 {
        return nil
    }
    timeout := recipes.HookTimeout(recipe, "initializeCommand")
    hctx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()
    g, gctx := errgroup.WithContext(hctx)
    for _, argv := range hook {
        argv := argv
        g.Go(func() error {
            cmd := exec.CommandContext(gctx, argv[0], argv[1:]...)
            out, err := cmd.CombinedOutput()
            if err != nil {
                return fmt.Errorf("host exec %v: %w (%s)", argv, err, string(out))
            }
            return nil
        })
    }
    return g.Wait()
}

func (r *Runner) teardown(ctx context.Context, cid string) error {
    tctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()
    _ = r.Stop(tctx, cid, nil)
    return r.Remove(tctx, cid)
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Recipe YAML lifecycle fields → hook normalizer → `docker exec` | The schema rejects non-string values inside hook argv; normalization preserves that invariant. |
| `initializeCommand` → host `os/exec` | Host-side escape hatch; v0.1 recipes leave this empty. The risk is a malicious recipe that declares `initializeCommand: ["rm", "-rf", "/"]`. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02.5-04 | Elevation of Privilege | In-container hook escape via shell metacharacters | mitigate | Hooks execute via `Runner.Exec(gctx, cid, argv)` where `argv []string` is passed directly to `moby/moby/client.ContainerExecCreate` — no shell interpolation. The only sh-c wrapping happens when the recipe author writes a bare string hook, which normalizes to `["sh","-c","<body>"]`; the body is the exact bytes the author committed (trusted via code review). Hooks run inside the Phase 2 sandbox posture: non-root `agent` user (post-gosu drop), `cap_drop=ALL`, `no_new_privs`, read-only rootfs + tmpfs, default seccomp. Plan 09's handler constructs RunOptions with those fields from `recipe.Isolation`. |
| T-02.5-04a | Elevation of Privilege | Host-side `initializeCommand` arbitrary execution | mitigate | v0.1 recipes have empty `initializeCommand`. `runHostInitialize` uses `os/exec.CommandContext` with the first argv element as the binary and the rest as args — no shell wrapping. Future Phase 4 recipes that declare a host-side initialize must pass code review. Considered and rejected: disallowing `initializeCommand` entirely — the Dev Containers spec uses it for GPU probes and Phase 4 will need it. Mitigation is code review of recipe.yaml, same as every other lifecycle hook. |
| T-02.5-02 | Information Disclosure | Secret leak during onCreate/updateContent install logs | mitigate | Plan 01's schema `hook-no-secrets` $def rejects any `secret:` reference inside `onCreateCommand` and `updateContentCommand`. Plan 01's semantic check also verifies every `secret:` ref in `auth.env` has a `secrets_schema` entry. The runner itself trusts the schema but runs hooks under the same sandbox posture as the agent process, so even if a recipe somehow smuggled a secret past the schema, the install log would still be routed through Plan 05's log redaction middleware. |
| T-02.5-05 | Denial of Service | Infinite loop hook blocks session forever | mitigate | Every hook runs under `context.WithTimeout(ctx, HookTimeout(recipe, name))`; default = 10 min; recipe can override up but not above a hard cap (none enforced in v0.1 — code review is the gate). Timeout triggers teardown + returns error. |
| T-02.5-06 | Denial of Service | errgroup parallel fan-out exhausts exec slots | mitigate | v0.1 recipes declare hooks by hand; the realistic max parallel fan-out is 3-4 groups per hook. Docker's default exec concurrency is sufficient. No explicit limit in 02.5; if a recipe ships with `postCreateCommand: [[...]]*100` it fails code review. |
