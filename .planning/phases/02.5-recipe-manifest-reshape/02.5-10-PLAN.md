**File 1 — test/smoke-matrix.sh:**

```bash
#!/usr/bin/env bash
# Gate A: smoke-test matrix for Phase 02.5 (D-46).
#
# Runs up to 4 cells: {aider, picoclaw} × {anthropic, openrouter}.
# Each cell: spawn session → send "whoareyou" → assert non-empty reply → teardown → assert no dangling container.
#
# Exit 0 if D-47 pass criteria are met (see bottom of script).
# Exit 1 if any non-SKIP cell fails OR D-47 criteria are not met.

set -uo pipefail

RECIPES=(aider picoclaw)
PROVIDERS=(anthropic openrouter)
API_URL="${API_URL:-http://127.0.0.1:8080}"
PROBE="whoareyou"

# ANSI colors for readability
C_GREEN=$'\e[32m'; C_RED=$'\e[31m'; C_YELLOW=$'\e[33m'; C_RESET=$'\e[0m'

# --- Teardown guards (global CLAUDE.md rule: kill previous before starting) ---
teardown_containers() {
    local stragglers
    stragglers=$(docker ps -aq --filter 'name=playground-' 2>/dev/null || true)
    if [ -n "$stragglers" ]; then
        echo "[teardown] removing $(echo "$stragglers" | wc -l | tr -d ' ') straggler container(s)..."
        docker rm -f $stragglers >/dev/null 2>&1 || true
    fi
}

echo "=== Gate A: smoke-test matrix ==="
echo "API: $API_URL"
echo "Probe: $PROBE"
echo

teardown_containers

# Check API is up
if ! curl -fsS --max-time 5 "$API_URL/healthz" >/dev/null 2>&1; then
    echo "${C_RED}FAIL${C_RESET}: API server not reachable at $API_URL"
    echo "       Start it in another terminal with: make run-api (or 'cd api && go run ./cmd/server/')"
    exit 1
fi

# --- Check BYOK keys ---
HAS_ANTHROPIC=0
HAS_OPENROUTER=0
[ -n "${AP_DEV_BYOK_KEY:-}" ] && HAS_ANTHROPIC=1
[ -n "${AP_DEV_OPENROUTER_KEY:-}" ] && HAS_OPENROUTER=1

echo "BYOK: anthropic=$HAS_ANTHROPIC openrouter=$HAS_OPENROUTER"
echo

# Authenticate (dev mode) to get a session cookie
COOKIE_JAR=$(mktemp)
trap 'rm -f "$COOKIE_JAR"; teardown_containers' EXIT
if ! curl -fsS -c "$COOKIE_JAR" -X POST "$API_URL/api/dev/login" >/dev/null 2>&1; then
    echo "${C_RED}FAIL${C_RESET}: could not authenticate via /api/dev/login (is AP_DEV_MODE=true?)"
    exit 1
fi

# --- Cell runner ---
declare -A RESULTS
PASS=0; FAIL=0; SKIP=0

run_cell() {
    local recipe="$1" provider="$2"
    local cell="${recipe}×${provider}"

    # Skip if BYOK key not set
    case "$provider" in
        anthropic)  [ "$HAS_ANTHROPIC"  -eq 0 ] && { echo "${C_YELLOW}[SKIP]${C_RESET} $cell — no AP_DEV_BYOK_KEY"; RESULTS[$cell]=SKIP; SKIP=$((SKIP+1)); return; } ;;
        openrouter) [ "$HAS_OPENROUTER" -eq 0 ] && { echo "${C_YELLOW}[SKIP]${C_RESET} $cell — no AP_DEV_OPENROUTER_KEY"; RESULTS[$cell]=SKIP; SKIP=$((SKIP+1)); return; } ;;
    esac

    # Check recipe declares this provider (D-46 behavior)
    local has_provider
    has_provider=$(curl -sf -b "$COOKIE_JAR" "$API_URL/api/recipes/$recipe" 2>/dev/null \
        | jq -r --arg p "$provider" '.providers[]? | select(.id == $p) | .id' 2>/dev/null)
    if [ -z "$has_provider" ]; then
        echo "${C_YELLOW}[SKIP]${C_RESET} $cell — recipe does not declare provider"
        RESULTS[$cell]=SKIP
        SKIP=$((SKIP+1))
        return
    fi

    # Pick first model matching this provider (D-48)
    local model
    model=$(curl -sf -b "$COOKIE_JAR" "$API_URL/api/recipes/$recipe" 2>/dev/null \
        | jq -r --arg p "$provider" '[.models[]? | select(.provider == $p)][0].id' 2>/dev/null)
    if [ -z "$model" ] || [ "$model" = "null" ]; then
        echo "${C_YELLOW}[SKIP]${C_RESET} $cell — no model declared for provider"
        RESULTS[$cell]=SKIP
        SKIP=$((SKIP+1))
        return
    fi

    printf '[RUN ] %s (model=%s) ...\n' "$cell" "$model"
    local t_start
    t_start=$(date +%s)

    # Spawn session
    local create_resp session_id
    create_resp=$(curl -sf -b "$COOKIE_JAR" -X POST "$API_URL/api/sessions" \
        -H 'content-type: application/json' \
        -d "{\"recipe\":\"$recipe\",\"provider\":\"$provider\",\"model\":\"$model\"}" 2>&1)
    session_id=$(echo "$create_resp" | jq -r '.id // empty' 2>/dev/null)
    if [ -z "$session_id" ] || [ "$session_id" = "null" ]; then
        echo "${C_RED}[FAIL]${C_RESET} $cell — session create failed: $create_resp"
        RESULTS[$cell]=FAIL
        FAIL=$((FAIL+1))
        return
    fi

    # Send whoareyou
    local msg_resp reply
    msg_resp=$(curl -sf -b "$COOKIE_JAR" -X POST "$API_URL/api/sessions/$session_id/message" \
        -H 'content-type: application/json' \
        -d "{\"text\":\"$PROBE\"}" 2>&1)
    reply=$(echo "$msg_resp" | jq -r '.text // .message // .reply // empty' 2>/dev/null)

    # Check for top-level error key (D-49)
    local has_error
    has_error=$(echo "$msg_resp" | jq -r '.error // empty' 2>/dev/null)

    # Teardown session
    curl -sf -b "$COOKIE_JAR" -X DELETE "$API_URL/api/sessions/$session_id" >/dev/null 2>&1 || true

    # Assert no dangling container
    local stragglers
    stragglers=$(docker ps -aq --filter "name=playground-" 2>/dev/null || true)
    if [ -n "$stragglers" ]; then
        echo "${C_RED}[FAIL]${C_RESET} $cell — dangling container(s) after DELETE: $stragglers"
        docker rm -f $stragglers >/dev/null 2>&1 || true
        RESULTS[$cell]=FAIL
        FAIL=$((FAIL+1))
        return
    fi

    local t_end elapsed
    t_end=$(date +%s)
    elapsed=$((t_end - t_start))

    if [ -n "$has_error" ]; then
        echo "${C_RED}[FAIL]${C_RESET} $cell — error envelope: $has_error (${elapsed}s)"
        RESULTS[$cell]=FAIL
        FAIL=$((FAIL+1))
        return
    fi

    if [ -z "$reply" ]; then
        echo "${C_RED}[FAIL]${C_RESET} $cell — empty reply (${elapsed}s) raw=$msg_resp"
        RESULTS[$cell]=FAIL
        FAIL=$((FAIL+1))
        return
    fi

    local snippet="${reply:0:60}"
    echo "${C_GREEN}[PASS]${C_RESET} $cell — ${elapsed}s → ${snippet}..."
    RESULTS[$cell]=PASS
    PASS=$((PASS+1))
}

# --- Run the matrix ---
for recipe in "${RECIPES[@]}"; do
    for provider in "${PROVIDERS[@]}"; do
        run_cell "$recipe" "$provider"
    done
done

# --- Print summary table ---
echo
echo "=== Matrix Summary ==="
printf '%-30s %s\n' "cell" "result"
for recipe in "${RECIPES[@]}"; do
    for provider in "${PROVIDERS[@]}"; do
        cell="${recipe}×${provider}"
        printf '%-30s %s\n' "$cell" "${RESULTS[$cell]:-MISSING}"
    done
done
echo "PASS=$PASS FAIL=$FAIL SKIP=$SKIP"
echo

# --- D-47 pass criteria enforcement ---
check_d47() {
    local rec
    # (1) Each recipe passes on at least one provider
    for rec in "${RECIPES[@]}"; do
        local ok=0
        for provider in "${PROVIDERS[@]}"; do
            if [ "${RESULTS[${rec}×${provider}]:-}" = "PASS" ]; then ok=1; break; fi
        done
        if [ "$ok" -eq 0 ]; then
            echo "${C_RED}D-47 FAIL${C_RESET}: $rec did not pass on any provider"
            return 1
        fi
    done
    # (2) At least one recipe passes on BOTH providers
    local both=0
    for rec in "${RECIPES[@]}"; do
        local count=0
        for provider in "${PROVIDERS[@]}"; do
            [ "${RESULTS[${rec}×${provider}]:-}" = "PASS" ] && count=$((count+1))
        done
        if [ "$count" -eq 2 ]; then both=1; break; fi
    done
    if [ "$both" -eq 0 ]; then
        echo "${C_RED}D-47 FAIL${C_RESET}: no recipe passes on both providers (OpenRouter wiring unproven)"
        return 1
    fi
    # (3) Zero non-SKIP failures
    if [ "$FAIL" -gt 0 ]; then
        echo "${C_RED}D-47 FAIL${C_RESET}: $FAIL non-SKIP failures"
        return 1
    fi
    return 0
}

if [ "$FAIL" -eq 0 ] && check_d47; then
    echo "${C_GREEN}=== Gate A PASSED ===${C_RESET}"
    exit 0
fi

echo "${C_RED}=== Gate A FAILED ===${C_RESET}"
exit 1
```

- Cookie jar handles auth via `/api/dev/login` (Phase 1 dev mode)
- Teardown guard runs on EXIT trap (covers normal + abnormal termination)
- D-47 enforcement is strict per CONTEXT.md

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `AP_DEV_BYOK_KEY` / `AP_DEV_OPENROUTER_KEY` env vars → API server → matrix script | Operator supplies real keys. Script reads them only to check "is this cell runnable". Keys never flow through the script to curl — they're consumed by the server via Plan 05's DevEnvSecretSource. |
| matrix script → curl → API server → docker → model provider | Full recipe path exercised. Any of the 10 plumbing steps from D-02b can fail here. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02.5-02 | Information Disclosure | BYOK key appears in curl output or in the failure message | mitigate | The script never sends BYOK keys via curl body. Keys are server-side state (DevEnvSecretSource reads env at API startup). Plan 05's log redaction writer scrubs any accidental server-log leak. The script prints only the reply snippet (first 60 chars) and error envelope — if the server accidentally echoes a key, the redaction writer fires first. |
| T-02.5-05 | Denial of Service | A cell hangs forever blocking the whole matrix | mitigate | Each curl has `--max-time 5` (health check) or implicit curl defaults. Each session has a recipe-declared `chat_io.response_timeout_sec` (60–120s). Docker container lifecycle has a fallback Stop timeout in runner.go. Total matrix wall time ≤ 4 cells × 120s = 480s worst case; target ≤ 90s typical. |
| T-02.5-06 | DoS / cleanup | Dangling containers consume host resources | mitigate | `teardown_containers` runs on EXIT trap (covers normal + signal termination) + explicit assertion after each DELETE. Any surviving container fails the cell and is force-removed before the next cell. Global CLAUDE.md rule enforced. |
