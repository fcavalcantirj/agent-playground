# Phase 02.5 rebuild: FROM ap-runtime-node instead of ap-base.

FROM ap-runtime-node:v0.1.0-22 AS build

# Minimal git install for the clone step (node:22-slim base lacks git).

RUN git clone https://github.com/sipeed/picoclaw.git picoclaw && \
    cd picoclaw && \
    git checkout "$PICOCLAW_SHA" && \
    npm ci --omit=dev && \
    npm run build || true && \
    ls -la

# Final stage: copy the built artifact on top of ap-runtime-node.
FROM ap-runtime-node:v0.1.0-22

COPY --from=build /build/picoclaw /opt/picoclaw

USER root
RUN ln -sf /opt/picoclaw/bin/picoclaw /usr/local/bin/picoclaw && \
    chown -R agent:agent /opt/picoclaw && \
    mkdir -p /home/agent/.picoclaw && \
    chown -R agent:agent /home/agent/.picoclaw

USER agent
WORKDIR /home/agent

# Inherit tini/tmux/ttyd/entrypoint shim from ap-runtime-node → ap-base. No ENTRYPOINT override.
