      ```go
      import (
          "regexp"
          "github.com/rs/zerolog"
      )

      var redactionPatterns = []*regexp.Regexp{
          regexp.MustCompile(`sk-ant-[A-Za-z0-9_-]{50,}`),
          regexp.MustCompile(`sk-or-v1-[a-f0-9]{64}`),
          regexp.MustCompile(`ap_session=[A-Za-z0-9+/=_-]+`),

      func (h RedactionHook) Run(e *zerolog.Event, level zerolog.Level, msg string) {
          // zerolog Events are write-only; the cleanest approach is to redact
          // via a custom FormattedMessage writer. In practice: use zerolog's
          // ConsoleWriter or inject via a wrapped io.Writer that rewrites lines
          // before they reach stdout. See InstallRedactionHook below.
          _ = e
          _ = level
          _ = msg
      }

      // InstallRedactionHook wraps an io.Writer so every line written through it
      // passes through the redaction regex set before reaching the underlying sink.
      // This is the PREFERRED approach — zerolog Hook.Run cannot mutate the event,
      // but wrapping the writer achieves the same result deterministically.
      func InstallRedactionHook(w io.Writer) io.Writer {
          return &redactingWriter{inner: w}
      }

      type redactingWriter struct{ inner io.Writer }

      func (r *redactingWriter) Write(p []byte) (int, error) {
          out := p
          for _, re := range redactionPatterns {
              out = re.ReplaceAll(out, []byte("<REDACTED>"))
          }
          if _, err := r.inner.Write(out); err != nil {
              return 0, err
          }
          return len(p), nil // report original length so zerolog internals don't panic
      }
      ```

```go
type DevEnvSecretSource struct {
    AnthropicKey  string
    OpenRouterKey string
    extras        map[string]string
}

type DevEnvSource = DevEnvSecretSource // Phase 2 alias

func NewDevEnvSource() *DevEnvSource {

}

func NewDevEnvSecretSource() *DevEnvSecretSource {
    s := &DevEnvSecretSource{
        AnthropicKey:  os.Getenv("AP_DEV_BYOK_KEY"),
        OpenRouterKey: os.Getenv("AP_DEV_OPENROUTER_KEY"),
        extras:        map[string]string{},
    }
    for _, kv := range os.Environ() {
        // AP_DEV_<NAME>_KEY=value, where NAME is all-caps alnum/underscore
        if !strings.HasPrefix(kv, "AP_DEV_") || !strings.Contains(kv, "_KEY=") {
            continue
        }
        eq := strings.IndexByte(kv, '=')
        name := kv[:eq]
        value := kv[eq+1:]
        if name == "AP_DEV_BYOK_KEY" || name == "AP_DEV_OPENROUTER_KEY" {
            continue
        }
        mid := strings.TrimSuffix(strings.TrimPrefix(name, "AP_DEV_"), "_KEY")
        normalized := strings.ToLower(strings.ReplaceAll(mid, "_", "-"))
        s.extras[normalized] = value
    }
    return s
}

func (s *DevEnvSecretSource) Get(name string) (string, error) {
    // Phase 2 implementation — keep as-is if already present. If missing, write it:
    switch normalizeSecretName(name) {
    case "anthropic-api-key":
        if s.AnthropicKey == "" {
            return "", ErrSecretMissing
        }
        return s.AnthropicKey, nil
    case "openrouter-api-key":
        if s.OpenRouterKey == "" {
            return "", ErrSecretMissing
        }
        return s.OpenRouterKey, nil
    default:
        if v, ok := s.extras[normalizeSecretName(name)]; ok {
            return v, nil
        }
        return "", ErrSecretMissing
    }
}

func (s *DevEnvSecretSource) Resolve(ref string) (string, error) {
    key, ok := strings.CutPrefix(ref, "secret:")
    if !ok {
        return ref, nil // literal passthrough
    }
    return s.Get(key)
}

// normalizeSecretName canonicalizes secret name variants to the hyphenated form.
// "anthropic_key" / "anthropic-key" / "anthropic_api_key" → "anthropic-api-key"
// "openrouter_key" / "openrouter-key" / "openrouter_api_key" → "openrouter-api-key"
func normalizeSecretName(name string) string {
    lower := strings.ToLower(strings.ReplaceAll(name, "_", "-"))
    switch lower {
    case "anthropic-key", "anthropic-api-key":
        return "anthropic-api-key"
    case "openrouter-key", "openrouter-api-key":
        return "openrouter-api-key"
    }
    return lower
}
```

```go
// Errors:
//   - ErrSecretMissing wrapped with field path on resolve failure
//   - template render errors wrapped with recipe ID + template name
//   - provider validation errors wrapped with "provider %q not declared"
func Materialize(
    ctx context.Context,
    recipe *recipes.Recipe,
    provider, modelID string,
    secrets SecretSource,
    templates *recipes.TemplateRegistry,
) (*MaterializedRecipe, error) {
    if recipe == nil {
        return nil, fmt.Errorf("materialize: nil recipe")
    }

    envOut := make(map[string]string, len(recipe.Auth.Env))
    for k, v := range recipe.Auth.Env {
        resolved, err := secrets.Resolve(v)
        if err != nil {
            return nil, fmt.Errorf("materialize env %q: %w", k, err)
        }
        envOut[k] = resolved
    }

    // Build template context: secrets map keyed by underscored name + env map + recipe
    secretCtx := make(map[string]string)
    for _, f := range recipe.Auth.Files {
        resolved, err := secrets.Resolve("secret:" + f.Secret)
        if err != nil {
            return nil, fmt.Errorf("materialize file secret %q: %w", f.Secret, err)
        }
        secretCtx[underscoreName(f.Secret)] = resolved
        secretCtx[f.Secret] = resolved // also keyed by original name
    }
    // Also expose env values under secrets.<envkey-lowercased> for template convenience
    for k, v := range envOut {
        secretCtx[underscoreName(strings.ToLower(k))] = v
    }

    data := map[string]any{
        "secrets": secretCtx,
        "env":     envOut,
        "recipe":  recipe,
    }

    var files []MaterializedFile
    for _, f := range recipe.Auth.Files {
        body, err := templates.Render(ctx, recipe.ID, f.Template, data)
        if err != nil {
            return nil, fmt.Errorf("materialize render %s/%s: %w", recipe.ID, f.Template, err)
        }
        mode := f.Mode
        if mode == "" {
            mode = "0600"
        }
        files = append(files, MaterializedFile{
            Target: f.Target,
            Mode:   mode,
            Body:   body,
        })
    }

    // Provider validation: ensure the requested provider is declared.
    if provider != "" {
        found := false
        for _, p := range recipe.Providers {
            if p.ID == provider {
                found = true
                break
            }
        }
        if !found {
            return nil, fmt.Errorf("materialize: provider %q not declared by recipe %q", provider, recipe.ID)
        }
    }

    return &MaterializedRecipe{
        Recipe: recipe,
        Env:    envOut,
        Files:  files,
    }, nil
}

// underscoreName converts "anthropic-api-key" → "anthropic_api_key" for
// text/template field access (hyphens are not valid in field names).
func underscoreName(s string) string {
    return strings.ReplaceAll(s, "-", "_")
}
```
