```python
"""age-based symmetric encryption for channel credentials at rest.

BYOK invariant: bot tokens, Telegram allowed-user IDs, and any future
channel-secret field arrive in a POST body, are held in memory for the
duration of the request, and are then either:
  (a) streamed into the container via --env-file (runner path — plaintext
      in RAM, 600-perm file, unlinked post-read), OR
  (b) persisted to agent_containers.channel_config_enc as an age-
      encrypted blob keyed by a per-user KEK derived from the server
      master key + user_id (this module).

The master key lives in env AP_CHANNEL_MASTER_KEY (32 bytes, base64).
It MUST be rotated via a dedicated re-encrypt migration; it MUST NOT
be logged. On boot the server fails loud if the env is missing AND
AGENT_PLAYGROUND_ENV is 'production'.

Per-user KEK derivation: HKDF-SHA256 over master_key with info=b'ap-ch-'
+ user_id.bytes. The age cipher then uses a deterministic scrypt
passphrase identity derived from the KEK.
"""
from __future__ import annotations

import base64
import json
import os
from uuid import UUID
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.hkdf import HKDF
import pyrage


def _master_key() -> bytes:
    raw = os.environ.get("AP_CHANNEL_MASTER_KEY")
    env = os.environ.get("AGENT_PLAYGROUND_ENV", "dev")
    if not raw:
        if env == "production":
            raise RuntimeError("AP_CHANNEL_MASTER_KEY required in production")
        # Deterministic dev fallback — 32 zero bytes. NEVER ship this value in prod.
        return b"\x00" * 32
    key = base64.b64decode(raw)
    if len(key) != 32:
        raise RuntimeError("AP_CHANNEL_MASTER_KEY must be 32 bytes (base64-encoded)")
    return key


def _derive_kek(user_id: UUID) -> bytes:
    hkdf = HKDF(
        algorithm=hashes.SHA256(),
        length=32,
        salt=None,
        info=b"ap-ch-" + user_id.bytes,
    )
    return hkdf.derive(_master_key())


def encrypt_channel_config(user_id: UUID, config: dict) -> bytes:
    """Encrypt a channel config dict (tokens + metadata) for at-rest storage."""
    kek = _derive_kek(user_id)
    passphrase = base64.b64encode(kek).decode("ascii")
    plaintext = json.dumps(config, separators=(",", ":")).encode("utf-8")
    return pyrage.passphrase.encrypt(plaintext, passphrase)


def decrypt_channel_config(user_id: UUID, ciphertext: bytes) -> dict:
    """Decrypt a stored channel config blob. Raises on MAC failure."""
    kek = _derive_kek(user_id)
    passphrase = base64.b64encode(kek).decode("ascii")
    plaintext = pyrage.passphrase.decrypt(ciphertext, passphrase)
    return json.loads(plaintext.decode("utf-8"))
```

```python
async def insert_pending_agent_container(
    conn: asyncpg.Connection,
    agent_instance_id: UUID,
    user_id: UUID,
    recipe_name: str,
    channel_type: str,
    channel_config_enc: bytes,
) -> UUID:
    """Insert a pending agent_containers row (status='starting').
    
    Returns the new row id. Enforces the partial unique index: if this
    agent_instance already has a running container, raises asyncpg.UniqueViolationError
    which the route maps to 409 AGENT_ALREADY_RUNNING.
    """
    # If a prior container is still 'starting', the partial index on
    # container_status='running' doesn't block us — but we still want
    # exclusive state. Conservative approach: do NOT serialize here,
    # let the route 409 if the partial index fires.
    row = await conn.fetchrow(
        """
        INSERT INTO agent_containers
            (id, agent_instance_id, user_id, recipe_name, deploy_mode,
             container_status, channel_type, channel_config_enc)
        VALUES (gen_random_uuid(), $1, $2, $3, 'persistent',
                'starting', $4, $5)
        RETURNING id
        """,
        agent_instance_id, user_id, recipe_name, channel_type, channel_config_enc,
    )
    return row["id"]


async def write_agent_container_running(
    conn: asyncpg.Connection,
    container_row_id: UUID,
    *,
    container_id: str,
    boot_wall_s: float,
    ready_at: datetime,
) -> None:
    """Mark a pending container row as running after execute_persistent_start succeeds.

    Writes container_id (docker short sha / cidfile content), boot_wall_s,
    ready_at, and flips container_status 'starting' -> 'running'. The
    partial unique index fires here: if a second concurrent /start hit
    this row, the flip to 'running' will throw UniqueViolation on the
    PK-covered index; the route treats that as a 409.
    """
    await conn.execute(
        """
        UPDATE agent_containers
           SET container_id = $2,
               container_status = 'running',
               boot_wall_s = $3,
               ready_at = $4
         WHERE id = $1
           AND container_status = 'starting'
        """,
        container_row_id, container_id, boot_wall_s, ready_at,
    )


async def mark_agent_container_stopped(
    conn: asyncpg.Connection,
    container_row_id: UUID,
    *,
    last_error: str | None = None,
) -> None:
    """Mark a running container as stopped. Called from /stop and from
    execute_persistent_start failure paths.

    Sets stopped_at=NOW(). When last_error is provided (error path), also
    writes container_status='start_failed' or 'crashed'. The caller decides
    which via the argument; this helper just persists the fields.
    """
    new_status = "start_failed" if last_error else "stopped"
    await conn.execute(
        """
        UPDATE agent_containers
           SET container_status = $2,
               stopped_at = NOW(),
               last_error = $3
         WHERE id = $1
        """,
        container_row_id, new_status, last_error,
    )


async def fetch_agent_container(
    conn: asyncpg.Connection,
    container_row_id: UUID,
) -> dict[str, Any] | None:
    """Fetch a single agent_containers row by id. Returns dict or None."""
    row = await conn.fetchrow(
        """
        SELECT id::text AS id,
               agent_instance_id::text AS agent_instance_id,
               user_id::text AS user_id,
               recipe_name, deploy_mode, container_id, container_status,
               channel_type, channel_config_enc,
               boot_wall_s, ready_at, created_at, stopped_at, last_error
          FROM agent_containers
         WHERE id = $1
        """,
        container_row_id,
    )
    if row is None:
        return None
    d = dict(row)
    if d.get("boot_wall_s") is not None:
        d["boot_wall_s"] = float(d["boot_wall_s"])
    return d


async def fetch_running_container_for_agent(
    conn: asyncpg.Connection,
    agent_instance_id: UUID,
) -> dict[str, Any] | None:
    """Fetch the currently-running container for an agent, or None.

    Used by /stop and /status routes. Relies on the partial unique index
    to guarantee at most one row can match.
    """
    row = await conn.fetchrow(
        """
        SELECT id::text AS id,
               agent_instance_id::text AS agent_instance_id,
               user_id::text AS user_id,
               recipe_name, container_id, container_status,
               channel_type, channel_config_enc,
               boot_wall_s, ready_at, created_at
          FROM agent_containers
         WHERE agent_instance_id = $1
           AND container_status = 'running'
         LIMIT 1
        """,
        agent_instance_id,
    )
    if row is None:
        return None
    d = dict(row)
    if d.get("boot_wall_s") is not None:
        d["boot_wall_s"] = float(d["boot_wall_s"])
    return d
```
