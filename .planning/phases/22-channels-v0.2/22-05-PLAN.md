async def fetch_agent_instance(
    conn: asyncpg.Connection,
    agent_id: UUID,
    user_id: UUID,
) -> dict[str, Any] | None:
    """Return the agent_instances row for (agent_id, user_id) or None.
    
    user_id scoping prevents cross-user reads once OAuth ships. For now the
    path is always ANONYMOUS_USER_ID; keeping the signature user-scoped so
    Phase 21 migration is a one-line change in the caller.
    """
    row = await conn.fetchrow(
        """
        SELECT id::text AS id,
               name, recipe_name, model, personality, created_at,
               last_run_at, total_runs
          FROM agent_instances
         WHERE id = $1 AND user_id = $2
        """,
        agent_id, user_id,
    )
    return dict(row) if row else None

```python
"""Persistent-mode agent lifecycle endpoints (Phase 22a).

Four endpoints:
  - POST /v1/agents/:id/start          spawn a persistent container
  - POST /v1/agents/:id/stop           graceful shutdown
  - GET  /v1/agents/:id/status         current container state
  - POST /v1/agents/:id/channels/:cid/pair   openclaw pairing approve

All four follow the 9-step flow from `routes/runs.py`. BYOK invariants
(data + log side) are identical — provider_key stays local, channel
creds flow into the age-encrypted blob for durable records + the
env-file for the actual container boot, and every exception string is
redacted before landing in the DB or the response.
"""
from __future__ import annotations

import logging
from datetime import datetime, timezone
from uuid import UUID

from fastapi import APIRouter, Header, Request
from fastapi.responses import JSONResponse

import asyncpg

from ..constants import ANONYMOUS_USER_ID
from ..crypto.age_cipher import encrypt_channel_config, decrypt_channel_config
from ..models.agents import (
    AgentStartRequest, AgentStartResponse,
    AgentStatusResponse, AgentStopResponse,
    AgentChannelPairRequest, AgentChannelPairResponse,
)
from ..models.errors import ErrorCode, make_error_envelope
from ..services.run_store import (
    fetch_agent_instance,
    insert_pending_agent_container, write_agent_container_running,
    mark_agent_container_stopped, fetch_agent_container,
    fetch_running_container_for_agent,
)
from ..services.runner_bridge import (
    execute_persistent_start, execute_persistent_stop,
    execute_persistent_status, execute_persistent_exec,
)
from ..util.ulid import new_run_id

router = APIRouter()
_log = logging.getLogger("api_server.agent_lifecycle")


def _err(status: int, code: str, message: str, *, param: str | None = None, category: str | None = None) -> JSONResponse:
    return JSONResponse(
        status_code=status,
        content=make_error_envelope(code, message, param=param, category=category),
    )


def _redact_creds(text: str, channel_inputs: dict[str, str]) -> str:
    """Replace every channel cred value with <REDACTED> in an error string."""
    out = text
    for var, val in channel_inputs.items():
        if val and len(val) >= 8:
            out = out.replace(val, "<REDACTED>")
        out = out.replace(f"{var}={val}", f"{var}=<REDACTED>") if val else out
    return out
```

POST /start handler (copy/adapt the 9-step flow):
```python
@router.post("/agents/{agent_id}/start")
async def start_agent(
    request: Request,
    agent_id: UUID,
    body: AgentStartRequest,
    authorization: str = Header(default=""),
):
    # --- Step 1: Bearer token (LLM api_key) ---
    if not authorization.startswith("Bearer "):
        return _err(401, ErrorCode.UNAUTHORIZED, "Bearer token required", param="Authorization")
    provider_key = authorization[len("Bearer "):].strip()
    if not provider_key:
        return _err(401, ErrorCode.UNAUTHORIZED, "Bearer token is empty", param="Authorization")

    # --- Step 2 + 3: resolve agent_instance ---
    pool = request.app.state.db
    async with pool.acquire() as conn:
        agent = await fetch_agent_instance(conn, agent_id, ANONYMOUS_USER_ID)
    if agent is None:
        return _err(404, ErrorCode.AGENT_NOT_FOUND, f"agent {agent_id} not found", param="agent_id")

    # --- Step 2b: validate recipe + channel + required inputs ---
    recipes = request.app.state.recipes
    recipe = recipes.get(agent["recipe_name"])
    if recipe is None:
        return _err(500, ErrorCode.INTERNAL, f"recipe {agent['recipe_name']!r} missing from server state")
    persistent = recipe.get("persistent")
    if not persistent:
        return _err(400, ErrorCode.CHANNEL_NOT_CONFIGURED,
                    f"recipe {agent['recipe_name']!r} has no persistent block", param="recipe")
    channels = recipe.get("channels") or {}
    channel = channels.get(body.channel)
    if channel is None:
        return _err(400, ErrorCode.CHANNEL_NOT_CONFIGURED,
                    f"recipe {agent['recipe_name']!r} does not support channel {body.channel!r}", param="channel")
    required = channel.get("required_user_input") or []
    missing = [e["env"] for e in required if e["env"] not in body.channel_inputs or not body.channel_inputs[e["env"]]]
    if missing:
        return _err(400, ErrorCode.CHANNEL_INPUTS_INVALID,
                    f"missing required channel inputs: {missing}", param="channel_inputs")
    api_key_var = recipe.get("runtime", {}).get("process_env", {}).get("api_key")
    if not api_key_var:
        return _err(500, ErrorCode.INTERNAL, "recipe missing runtime.process_env.api_key")

    # --- Step 4 + 5: encrypt creds + insert pending container row ---
    # DB scope 1 — opens + closes BEFORE the long await.
    config_plain = {"channel": body.channel, "inputs": dict(body.channel_inputs)}
    try:
        config_enc = encrypt_channel_config(ANONYMOUS_USER_ID, config_plain)
    except Exception as e:
        _log.error("channel config encryption failed", extra={"agent_id": str(agent_id)})
        return _err(500, ErrorCode.INTERNAL, "channel config encryption failed")

    try:
        async with pool.acquire() as conn:
            try:
                container_row_id = await insert_pending_agent_container(
                    conn, agent_id, ANONYMOUS_USER_ID, agent["recipe_name"], body.channel, config_enc,
                )
            except asyncpg.UniqueViolationError:
                return _err(409, ErrorCode.AGENT_ALREADY_RUNNING,
                            f"agent {agent_id} already has a running container", param="agent_id")
    except Exception:
        _log.exception("insert_pending_agent_container failed", extra={"agent_id": str(agent_id)})
        return _err(500, ErrorCode.INTERNAL, "failed to persist pending container row")

    # --- Step 7: execute_persistent_start (no DB held) ---
    run_id = new_run_id()
    boot_timeout_s = body.boot_timeout_s or 180
    try:
        details = await execute_persistent_start(
            request.app.state,
            recipe,
            model=agent["model"],
            api_key_var=api_key_var,
            api_key_val=provider_key,
            channel_id=body.channel,
            channel_creds=dict(body.channel_inputs),
            run_id=run_id,
            boot_timeout_s=boot_timeout_s,
        )
    except Exception as e:
        redacted = str(e).replace(provider_key, "<REDACTED>")
        redacted = _redact_creds(redacted, body.channel_inputs)
        async with pool.acquire() as conn:
            await mark_agent_container_stopped(conn, container_row_id, last_error=redacted[:500])
        _log.error("execute_persistent_start failed", extra={"agent_id": str(agent_id)})
        return _err(502, ErrorCode.INFRA_UNAVAILABLE, "runner failed to start container", category="INFRA_FAIL")

    # --- Step 8: verify PASS and complete row ---
    if details.get("verdict") != "PASS":
        # Container failed to boot: INVOKE_FAIL or TIMEOUT; details["detail"] is already
        # redacted by run_cell_persistent.
        detail = details.get("detail") or f"verdict={details.get('verdict')}"
        redacted = _redact_creds(detail, body.channel_inputs)
        async with pool.acquire() as conn:
            await mark_agent_container_stopped(conn, container_row_id, last_error=redacted[:500])
        return _err(502, ErrorCode.INFRA_UNAVAILABLE,
                    f"container failed to reach ready: {redacted[:200]}",
                    category=details.get("category") or "INFRA_FAIL")

    ready_at = datetime.now(timezone.utc)
    container_id = details["container_id"]
    boot_wall_s = float(details.get("boot_wall_s") or 0.0)
    async with pool.acquire() as conn:
        try:
            await write_agent_container_running(
                conn, container_row_id,
                container_id=container_id, boot_wall_s=boot_wall_s, ready_at=ready_at,
            )
        except asyncpg.UniqueViolationError:
            # Extremely unlikely: race against another /start that also won the semaphore.
            # Kill the newly-booted container to avoid orphaning it.
            await execute_persistent_stop(container_id, graceful_shutdown_s=3,
                                          data_dir=details.get("data_dir"))
            return _err(409, ErrorCode.AGENT_ALREADY_RUNNING,
                        f"agent {agent_id} already has a running container")

    # --- Step 9: response ---
    return AgentStartResponse(
        agent_id=agent_id,
        container_row_id=container_row_id,
        container_id=container_id,
        container_status="running",
        channel=body.channel,
        ready_at=ready_at,
        boot_wall_s=boot_wall_s,
        health_check_ok=bool(details.get("health_check_ok")),
        health_check_kind=str(details.get("health_check_kind") or "unknown"),
    ).model_dump(mode="json")
```

Gotchas:
- `provider_key` is the LLM key (OpenRouter / Anthropic / etc.) — same discipline as `/v1/runs`: local var, never logged, redacted on exception.
- `body.channel_inputs` is a dict[str, str] — no nesting. Serialize to JSON-safe plaintext BEFORE age-encrypt (json.dumps in the crypto helper handles this).
- UniqueViolationError can fire at insert (if a row for this agent is already in 'running' state) OR at UPDATE→running (if two starts race past the pending insert). Catch both.
- `last_error` from mark_agent_container_stopped must be the REDACTED string — never the raw exception.
- `provider_key` redaction + channel_inputs redaction: call both. Provider key first (longer + more sensitive), channel creds second.
- Do NOT persist `body.channel_inputs` plaintext ANYWHERE. The age-encrypted blob in `channel_config_enc` is the ONLY durable copy; and even that is only read again if Plan 23 wants to restart the container without the user re-entering creds.
- `decrypt_channel_config` is imported but NOT called in this plan — Plan 23 will use it for restart-with-stored-creds. Its import is intentional so the import path is proven correct.

@router.post("/agents/{agent_id}/channels/{cid}/pair")
async def pair_channel(
    request: Request,
    agent_id: UUID,
    cid: str,
    body: AgentChannelPairRequest,
    authorization: str = Header(default=""),
):
    if not authorization.startswith("Bearer "):
        return _err(401, ErrorCode.UNAUTHORIZED, "Bearer token required", param="Authorization")

    pool = request.app.state.db
    async with pool.acquire() as conn:
        agent = await fetch_agent_instance(conn, agent_id, ANONYMOUS_USER_ID)
        if agent is None:
            return _err(404, ErrorCode.AGENT_NOT_FOUND, f"agent {agent_id} not found", param="agent_id")
        running = await fetch_running_container_for_agent(conn, agent_id)
    if running is None:
        return _err(409, ErrorCode.AGENT_NOT_RUNNING,
                    f"agent {agent_id} has no running container", param="agent_id")

    recipe = request.app.state.recipes.get(agent["recipe_name"])
    if recipe is None:
        return _err(500, ErrorCode.INTERNAL, f"recipe {agent['recipe_name']!r} missing from server state")

    channel = (recipe.get("channels") or {}).get(cid)
    if channel is None:
        return _err(400, ErrorCode.CHANNEL_NOT_CONFIGURED,
                    f"recipe {agent['recipe_name']!r} does not support channel {cid!r}", param="cid")
    pairing = channel.get("pairing") or {}
    approve_argv = pairing.get("approve_argv")
    if not approve_argv:
        return _err(400, ErrorCode.CHANNEL_NOT_CONFIGURED,
                    f"channel {cid} on recipe {agent['recipe_name']!r} has no pairing.approve_argv",
                    param="cid")

    # substitute $CODE in the argv list
    argv = [arg.replace("$CODE", body.code) for arg in approve_argv]

    try:
        details = await execute_persistent_exec(
            request.app.state,
            running["container_id"],
            argv,
            timeout_s=30,
        )
    except Exception:
        _log.exception("pair exec failed", extra={"agent_id": str(agent_id)})
        return _err(502, ErrorCode.INFRA_UNAVAILABLE, "pair command failed", category="INFRA_FAIL")

    return AgentChannelPairResponse(
        agent_id=agent_id,
        channel=cid,
        exit_code=int(details.get("exit_code") or -1),
        stdout_tail=str(details.get("stdout_tail") or ""),
        stderr_tail=str(details.get("stderr_tail") or ""),
        wall_time_s=float(details.get("wall_time_s") or 0.0),
    ).model_dump(mode="json")
