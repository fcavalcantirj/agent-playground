```python
def _import_run_recipe_module():
    """Import and cache the run_recipe module (full module, not just one callable).

    All persistent-mode runner functions (run_cell, run_cell_persistent,
    stop_persistent, exec_in_persistent) live in the same file. Caching the
    module object in sys.modules["run_recipe"] keeps Verdict/Category identity
    stable across lint_service + execute_run + execute_persistent_* callers.
    """
    mod = sys.modules.get("run_recipe")
    if mod is not None and hasattr(mod, "run_cell"):
        return mod
    repo_root = Path(__file__).resolve().parents[4]
    runner_path = repo_root / "tools" / "run_recipe.py"
    if not runner_path.exists():
        raise RuntimeError(f"runner not found at {runner_path}")
    spec = importlib.util.spec_from_file_location("run_recipe", runner_path)
    if spec is None or spec.loader is None:
        raise RuntimeError(f"failed to build import spec for {runner_path}")
    new_mod = importlib.util.module_from_spec(spec)
    sys.modules["run_recipe"] = new_mod
    spec.loader.exec_module(new_mod)
    return new_mod


def _import_run_cell():
    """Back-compat shim used by execute_run."""
    return _import_run_recipe_module().run_cell
```

```python
async def execute_persistent_start(
    app_state,
    recipe: dict,
    *,
    model: str,
    api_key_var: str,
    api_key_val: str,
    channel_id: str,
    channel_creds: dict[str, str],
    run_id: str,
    boot_timeout_s: int = 180,
) -> dict[str, Any]:
    """Spawn a persistent container using the same concurrency scaffold as execute_run.

    Per-tag Lock serializes image builds (first call on a cold cache builds/
    pulls; subsequent concurrent starts share the cached image). Global
    Semaphore bounds total in-flight boots across all recipes.

    Returns the details dict half of run_cell_persistent's return. On non-PASS
    verdict the caller (route handler) decides whether to raise or record
    the failure; this function returns the dict unconditionally.
    """
    mod = _import_run_recipe_module()
    image_tag = f"ap-recipe-{recipe['name']}"
    tag_lock = await _get_tag_lock(app_state, image_tag)
    async with tag_lock:
        async with app_state.run_semaphore:
            result = await asyncio.to_thread(
                mod.run_cell_persistent,
                recipe,
                image_tag=image_tag,
                model=model,
                api_key_var=api_key_var,
                api_key_val=api_key_val,
                channel_id=channel_id,
                channel_creds=channel_creds,
                run_id=run_id,
                quiet=True,
                boot_timeout_s=boot_timeout_s,
            )
    # Same tuple / dict shape handling as execute_run (tests short-circuit).
    if isinstance(result, tuple):
        verdict, details = result
        # Attach verdict string so route layer can branch without introspecting
        # the Verdict object (which is a bespoke namedtuple defined in the
        # runner — keep it internal).
        details = dict(details)
        details["verdict"] = verdict.verdict
        details["category"] = verdict.category.value if hasattr(verdict.category, "value") else verdict.category
        details["detail"] = verdict.detail
    else:
        details = dict(result)
        details.setdefault("verdict", "PASS")
    return details
```

Gotchas:
- `tag_lock` is held for the duration of boot (~10-120s depending on recipe). That's intentional: concurrent `/start` requests for the same recipe serialize on image cache. For different recipes, they run in parallel (bounded by the semaphore).
- DO NOT log channel_creds or api_key_val in this module. The call chain below (Plan 22-05) redacts exceptions before persistence.
- Returning `dict` (not the tuple) matches `execute_run`'s contract — route layer reads `details["verdict"]` string + `details["category"]` string.
- `boot_timeout_s=180` default covers openclaw's 120s boot with headroom. Route layer can bump for Dockerfile cold-build cases.

```python
async def execute_persistent_stop(
    container_id: str,
    *,
    graceful_shutdown_s: int = 5,
    data_dir: str | None = None,
) -> dict[str, Any]:
    """Gracefully stop a persistent container.

    No image lock needed (stop doesn't touch images). No semaphore (stop is
    cheap and concurrent-safe). Still wraps in to_thread so the sync
    subprocess calls don't stall the event loop.
    """
    mod = _import_run_recipe_module()
    result = await asyncio.to_thread(
        mod.stop_persistent,
        container_id,
        graceful_shutdown_s=graceful_shutdown_s,
        data_dir=data_dir,
        quiet=True,
    )
    if isinstance(result, tuple):
        verdict, details = result
        details = dict(details)
        details["verdict"] = verdict.verdict
    else:
        details = dict(result)
        details.setdefault("verdict", "PASS")
    return details


async def execute_persistent_status(
    container_id: str,
    *,
    log_tail_lines: int = 50,
) -> dict[str, Any]:
    """Return current state of a persistent container.

    Shape: {running: bool, exit_code: int | None, log_tail: list[str]}.
    Read-only — no lock, no semaphore. Uses subprocess to invoke docker
    inspect + docker logs. Factored here (not as a new run_recipe
    function) because it's API-layer concern, not runner primitive.
    """
    def _probe() -> dict[str, Any]:
        import subprocess as _sp
        running = False
        exit_code: int | None = None
        inspect = _sp.run(
            ["docker", "inspect", "-f", "{{.State.Running}},{{.State.ExitCode}}", container_id],
            timeout=5, capture_output=True, text=True, check=False,
        )
        if inspect.returncode == 0:
            parts = (inspect.stdout or "").strip().split(",", 1)
            running = (parts[0].strip() == "true")
            try:
                exit_code = int(parts[1].strip()) if len(parts) > 1 else None
            except ValueError:
                exit_code = None
        logs = _sp.run(
            ["docker", "logs", "--tail", str(log_tail_lines), container_id],
            timeout=10, capture_output=True, text=True, check=False,
        )
        log_tail = ((logs.stdout or "") + "\n" + (logs.stderr or "")).strip().splitlines()[-log_tail_lines:]
        return {"running": running, "exit_code": exit_code, "log_tail": log_tail}
    return await asyncio.to_thread(_probe)


async def execute_persistent_exec(
    app_state,
    container_id: str,
    argv: list[str],
    *,
    timeout_s: int = 30,
) -> dict[str, Any]:
    """Run a command inside a persistent container (openclaw pairing approve, etc.).

    Uses the global semaphore to avoid flooding the docker daemon with
    concurrent execs across unrelated containers. No image lock (exec doesn't
    touch images).
    """
    mod = _import_run_recipe_module()
    async with app_state.run_semaphore:
        result = await asyncio.to_thread(
            mod.exec_in_persistent,
            container_id,
            argv,
            timeout_s=timeout_s,
        )
    if isinstance(result, tuple):
        verdict, details = result
        details = dict(details)
        details["verdict"] = verdict.verdict
    else:
        details = dict(result)
        details.setdefault("verdict", "PASS")
    return details
```

Gotchas:
- `execute_persistent_status` does NOT import from `run_recipe` — it's self-contained subprocess calls because the probe logic is trivial and adding it to `run_recipe.py` would be noise. Documented in-line.
- Semaphore on `_exec` but NOT on `_stop` or `_status`: exec runs agent CLI commands that may be slow (openclaw pairing approve is sub-second but other exec paths may be slower); stop/status are millisecond-scale probes.
- No lock on any of these three: they operate on an EXISTING container, image state is irrelevant.
- `asyncio.to_thread` is mandatory for all subprocess calls — even the "fast" docker inspect in _status can block if the daemon is slow.
