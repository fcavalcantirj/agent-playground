```python
"""Phase 22c.3 — in-app chat channel HTTP handlers.

Three endpoints:
  - POST   /v1/agents/:id/messages          — fast-ack inbound (D-07, D-29)
  - GET    /v1/agents/:id/messages/stream   — SSE outbound (D-01, D-25, D-26)  [Task 3]
  - DELETE /v1/agents/:id/messages          — clear conversation history (D-43, D-44)

Auth: every handler uses require_user (D-18, D-19) — `ap_session` cookie
mandatory; ownership filtered at SQL via fetch_agent_instance.
"""
from __future__ import annotations
import json
import logging
import re
from datetime import datetime, timezone
from uuid import UUID

import asyncpg
from fastapi import APIRouter, Header, Request
from fastapi.responses import JSONResponse, Response
from pydantic import BaseModel, Field

from ..auth.deps import require_user
from ..models.errors import ErrorCode, make_error_envelope
from ..models.events import VALID_KINDS
from ..services import inapp_messages_store as ims
from ..services.run_store import fetch_agent_instance

router = APIRouter()
_log = logging.getLogger("api_server.agent_messages")

INAPP_KINDS = ("inapp_inbound", "inapp_outbound", "inapp_outbound_failed")
# BYOK leak defense (CONTEXT.md anti-pattern: chat content might carry an API key).
_BYOK_LEAK_RE = re.compile(
    r"(OPENROUTER_API_KEY|ANTHROPIC_API_KEY|sk-[a-zA-Z0-9]{40,})"
)


def _err(status, code, message, *, param=None, category=None):
    return JSONResponse(
        status_code=status,
        content=make_error_envelope(code, message, param=param, category=category),
    )


class PostMessageRequest(BaseModel):
    content: str = Field(..., min_length=1)
    # D-39: Idempotency-Key is sent as HTTP header by middleware; not in body.


class PostMessageResponse(BaseModel):
    message_id: str
    status: str
    queued_at: datetime


@router.post("/agents/{agent_id}/messages", status_code=202)
async def post_message(
    request: Request,
    agent_id: UUID,
    body: PostMessageRequest,
):
    """D-07, D-29: fast-ack 202 with {message_id}. Dispatcher picks the row."""
    # --- Step 1: require_user (D-18) ---
    sess = require_user(request)
    if isinstance(sess, JSONResponse):
        return sess
    user_id: UUID = sess

    # --- Step 2: ownership (D-19) ---
    pool = request.app.state.db
    async with pool.acquire() as conn:
        agent = await fetch_agent_instance(conn, agent_id, user_id)
    if agent is None:
        return _err(404, ErrorCode.AGENT_NOT_FOUND,
                    f"agent {agent_id} not found", param="agent_id")

    # --- Step 2b: BYOK leak defense (CONTEXT anti-pattern) ---
    if _BYOK_LEAK_RE.search(body.content):
        return _err(400, ErrorCode.INVALID_REQUEST,
                    "content appears to contain an API key — refusing to persist",
                    param="content")

    # --- Step 3: INSERT pending ---
    async with pool.acquire() as conn:
        message_id = await ims.insert_pending(
            conn, agent_id=agent_id, user_id=user_id, content=body.content,
        )

    return JSONResponse(
        status_code=202,
        content={
            "message_id": str(message_id),
            "status": "pending",
            "queued_at": datetime.now(timezone.utc).isoformat(),
        },
    )


@router.delete("/agents/{agent_id}/messages")
async def delete_history(request: Request, agent_id: UUID):
    """D-43, D-44: clear inapp_messages + agent_events (inapp_* kinds) atomically.

    Container UNTOUCHED. Different from DELETE /v1/agents/:id (which
    cascades container teardown). Returns 204.
    """
    sess = require_user(request)
    if isinstance(sess, JSONResponse):
        return sess
    user_id: UUID = sess

    pool = request.app.state.db
    async with pool.acquire() as conn:
```

```python
import asyncio
import json
import logging
from sse_starlette.sse import EventSourceResponse, ServerSentEvent

REPLAY_HARD_CAP = 500   # D-26
SSE_PING_S = 30          # Pitfall 8 (Cloudflare/middlebox idle)


@router.get("/agents/{agent_id}/messages/stream")
async def messages_stream(
    request: Request,
    agent_id: UUID,
    last_event_id: str | None = Header(default=None, alias="Last-Event-Id"),
):
    """SSE outbound stream — D-01, D-25, D-26.

    URL contract: ``agent_id`` IS ``agent_instances.id`` for ALL Phase 22c.3
    endpoints (POST /messages, GET /messages/stream, DELETE /messages).
    This DIFFERS from the Phase 22b ``GET /v1/agents/:id/events`` route
    which uses ``agent_containers.id`` — that route's contract is
    unchanged. The Phase 22c.3 router resolves agent_instances.id ->
    most-recent agent_containers.id internally before querying agent_events.

    1. require_user + fetch_agent_instance (ownership at SQL).
    2. Resolve container_row_id: SELECT id FROM agent_containers WHERE
       agent_instance_id=$1 AND stopped_at IS NULL ORDER BY created_at DESC
       LIMIT 1 (prefer running container; 404 if none).
    3. Replay agent_events seq > Last-Event-Id (capped at 500; emit
       replay_truncated event when over).
    4. Capture current_max_seq AFTER replay (Pitfall 1 race window).
    5. Subscribe to redis channel agent:inapp:<agent_instance_id>.
    6. Differential second replay for ``seq > last_yielded`` (live consumer
       de-duplicates via last_yielded_seq guard).
    7. Live-stream from Redis; ping every 30s.
    """
    sess = require_user(request)
    if isinstance(sess, JSONResponse):
        return sess
    user_id: UUID = sess

    pool = request.app.state.db
    async with pool.acquire() as conn:
        agent = await fetch_agent_instance(conn, agent_id, user_id)
    if agent is None:
        return _err(404, ErrorCode.AGENT_NOT_FOUND,
                    f"agent {agent_id} not found", param="agent_id")

    # Parse Last-Event-Id (D-25)
    since_seq = 0
    if last_event_id:
        try:
            since_seq = max(0, int(last_event_id))
        except ValueError:
            since_seq = 0

    redis_client = request.app.state.redis
    # D-09 channel naming convention: agent:inapp:<agent_instance_id>.
    # URL agent_id IS agent_instances.id for Phase 22c.3 endpoints.
    channel = f"agent:inapp:{agent_id}"
    inapp_kinds = set(INAPP_KINDS)

    # Phase 22c.3 URL contract: resolve agent_instances.id ->
    # most-recent agent_containers.id BEFORE querying agent_events
    # (which is keyed on agent_container_id). Prefer a running container;
    # fall back to most-recently-created. 404 if no container row exists.
    async with pool.acquire() as conn:
        container_row_id = await conn.fetchval(
            """
            SELECT id FROM agent_containers
            WHERE agent_instance_id=$1
            ORDER BY (stopped_at IS NULL) DESC, created_at DESC
            LIMIT 1
            """,
            agent_id,
        )
    if container_row_id is None:
        return _err(404, ErrorCode.AGENT_NOT_FOUND,
                    f"agent {agent_id} has no container",
                    param="agent_id")

    async def event_generator():
        from ..services.event_store import fetch_events_after_seq

        # ----- Phase 1: PG replay (DB scope released at end) -----
        # fetch_events_after_seq is keyed on agent_container_id (= the
        # row id of the most-recent agent_containers row resolved above).
        async with pool.acquire() as conn:
            rows = await fetch_events_after_seq(
                conn, container_row_id, since_seq, inapp_kinds,
            )
            current_max_seq = await conn.fetchval(
                "SELECT COALESCE(MAX(seq), 0) FROM agent_events "
                "WHERE agent_container_id=$1 AND kind = ANY($2::text[])",
                container_row_id, list(INAPP_KINDS),
            )
        truncated = False
        if len(rows) > REPLAY_HARD_CAP:
            rows = rows[:REPLAY_HARD_CAP]
            truncated = True

        last_yielded_seq = since_seq
        for r in rows:
            payload = r["payload"]
            if isinstance(payload, str):
                payload = json.loads(payload)
            seq_int = int(r["seq"])
            yield ServerSentEvent(
                id=str(seq_int),
                event=r["kind"],
                data=json.dumps({
                    "seq": seq_int,
                    "kind": r["kind"],
                    "payload": payload,
                    "correlation_id": r.get("correlation_id"),
                    "ts": (r["ts"].isoformat() if hasattr(r["ts"], "isoformat") else str(r["ts"])),
                }),
            )
            last_yielded_seq = max(last_yielded_seq, seq_int)
        if truncated:
            # D-26: emit informational event; stream STAYS OPEN. Client uses
            # GET /v1/agents/:id/events?kinds=inapp_*&since_seq=... to fetch
            # the events seq <= (since_seq + 500) that were trimmed off.
            yield ServerSentEvent(
                event="replay_truncated",
                data=json.dumps({
                    "hint": "stream continues; backfill trimmed range via GET /v1/agents/:id/events",
                    "trimmed_above_seq": int(rows[-1]["seq"]) if rows else since_seq,
                }),
            )

        # ----- Phase 2: Subscribe + differential replay -----
        async with redis_client.pubsub() as pubsub:
            await pubsub.subscribe(channel)

            # Pitfall 1 mitigation: events INSERTed between phase 1's
            # query and the subscribe attach are LIVE-missable. Run
            # ANOTHER PG query for seq > last_yielded AND seq <= current_max.
            # current_max was captured BEFORE subscribe so we know the
            # window. Then attach the live consumer.
            async with pool.acquire() as conn:
                gap_rows = await fetch_events_after_seq(
                    conn, container_row_id, last_yielded_seq, inapp_kinds,
                )
            # Emit ALL gap_rows; the live consumer below de-duplicates via
            # last_yielded_seq tracking against subsequent live Redis messages
            # carrying the same seq.
            for r in gap_rows:
                seq_int = int(r["seq"])
                payload = r["payload"]
                if isinstance(payload, str):
                    payload = json.loads(payload)
                yield ServerSentEvent(
                    id=str(seq_int), event=r["kind"],
                    data=json.dumps({
                        "seq": seq_int, "kind": r["kind"], "payload": payload,
                        "correlation_id": r.get("correlation_id"),
                        "ts": r["ts"].isoformat(),
                    }),
                )
                last_yielded_seq = max(last_yielded_seq, seq_int)

            # ----- Phase 3: live consumer -----
            while True:
                if await request.is_disconnected():
                    return
                try:
                    msg = await pubsub.get_message(
                        ignore_subscribe_messages=True, timeout=1.0,
                    )
                except Exception:
                    _log.exception("phase22c3.sse.pubsub_get_message_failed")
                    # Defensive re-subscribe + PG catch-up (Pitfall 4 redux).
                    try:
                        await pubsub.unsubscribe(channel)
                    except Exception:
                        pass
                    await pubsub.subscribe(channel)
                    async with pool.acquire() as conn:
                        catch_up = await fetch_events_after_seq(
                            conn, container_row_id, last_yielded_seq, inapp_kinds,
                        )
                    for r in catch_up:
                        seq_int = int(r["seq"])
                        payload = r["payload"]
                        if isinstance(payload, str):
                            payload = json.loads(payload)
                        yield ServerSentEvent(
                            id=str(seq_int), event=r["kind"],
                            data=json.dumps({
                                "seq": seq_int, "kind": r["kind"], "payload": payload,
                                "correlation_id": r.get("correlation_id"),
                                "ts": r["ts"].isoformat(),
                            }),
                        )
                        last_yielded_seq = max(last_yielded_seq, seq_int)
                    continue
                if msg is None:
                    continue
                data = msg["data"]
                if isinstance(data, (bytes, bytearray)):
                    data = data.decode("utf-8")
                try:
                    parsed = json.loads(data)
                    seq_int = int(parsed["seq"])
                    if seq_int <= last_yielded_seq:
                        continue  # already yielded via differential replay
                    yield ServerSentEvent(
                        id=str(seq_int),
                        event=parsed["kind"],
                        data=data,
                    )
                    last_yielded_seq = seq_int
                except Exception:
                    _log.exception("phase22c3.sse.malformed_redis_message")
                    continue

    return EventSourceResponse(event_generator(), ping=SSE_PING_S)
```
