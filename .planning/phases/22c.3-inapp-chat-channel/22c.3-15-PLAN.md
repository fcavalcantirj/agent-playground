```python
import pytest, json, time, asyncio, pathlib

RECIPE_MATRIX = [
    ("hermes",   "openai_compat",   8642,  "/v1/chat/completions"),
    ("nanobot",  "openai_compat",   8900,  "/v1/chat/completions"),
    ("openclaw", "openai_compat",   18789, "/v1/chat/completions"),
    ("nullclaw", "a2a_jsonrpc",     3000,  "/a2a"),
    ("zeroclaw", "zeroclaw_native", 42617, "/webhook"),
]

REPORT_PATH = pathlib.Path(__file__).parent / "e2e-report.json"

@pytest.fixture(scope="session")
def report_accumulator():
    return {"passed": True, "recipes": [], "failures": []}

@pytest.fixture(scope="session", autouse=True)
def emit_report(report_accumulator):
    yield
    REPORT_PATH.write_text(json.dumps(report_accumulator, indent=2))

@pytest.mark.parametrize("recipe,contract,port,endpoint", RECIPE_MATRIX,
                         ids=[r[0] for r in RECIPE_MATRIX])
async def test_recipe_inapp_round_trip(
    recipe, contract, port, endpoint,
    started_api_server, fake_oauth_user_with_openrouter_key,
    recipe_container_factory, report_accumulator,
):
    t0 = time.monotonic()
    container = await recipe_container_factory(recipe)
    user_id = fake_oauth_user_with_openrouter_key
    api_url = started_api_server

    try:
        agent_id = await _ensure_agent_for_recipe(api_url, user_id, recipe, contract, port, endpoint)
        message_id = await post_inapp_message(api_url, user_id, agent_id, "who are you in 1 short sentence?")
        result = await wait_for_sse_event(api_url, user_id, agent_id, message_id, timeout_s=600)
        latency_ms = int((time.monotonic() - t0) * 1000)

        assert result["status"] == "done", f"expected done, got {result['status']}"
        assert len(result["bot_response"]) > 0, "empty bot_response"
        await assert_status_transitions(api_url, message_id, ["pending","forwarded","done"])

        report_accumulator["recipes"].append({
            "recipe": recipe, "contract": contract,
            "latency_ms": latency_ms,
            "bot_response_excerpt": result["bot_response"][:200],
            "status": "PASS",
        })
    except Exception as exc:
        report_accumulator["passed"] = False
        report_accumulator["recipes"].append({
            "recipe": recipe, "contract": contract,
            "status": "FAIL", "error": repr(exc),
        })
        report_accumulator["failures"].append({"recipe": recipe, "error": repr(exc)})
        raise
```
