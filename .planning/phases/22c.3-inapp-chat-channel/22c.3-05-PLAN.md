Per D-32 (`FOR UPDATE SKIP LOCKED`): multi-replica safe today; single-replica today. Per D-37/D-38 (readiness gate): unready containers fail-fast at fetch time, no wasted bot calls.

async def fetch_pending_for_dispatch(conn, limit: int) -> list[asyncpg.Record]:
    """Returns rows JOINed with agent_containers; columns:
       m.id, m.agent_id, m.user_id, m.content, m.attempts,
       c.id AS container_row_id, c.container_id, c.container_status,
       c.ready_at, c.stopped_at, c.recipe_name, c.channel_type, c.inapp_auth_token
       FOR UPDATE OF m SKIP LOCKED. Caller MUST be in a transaction."""

async def mark_forwarded(conn, message_id: UUID) -> None:
    """UPDATE inapp_messages SET status='forwarded', attempts=attempts+1, last_attempt_at=NOW()
       WHERE id=$1 AND status='pending'."""

async def mark_done(conn, message_id: UUID, bot_response: str) -> None:
    """UPDATE inapp_messages SET status='done', bot_response=$2, completed_at=NOW()
       WHERE id=$1 AND status='forwarded'."""

async def mark_failed(conn, message_id: UUID, last_error: str) -> None:
    """UPDATE inapp_messages SET status='failed', last_error=$2, completed_at=NOW()
       WHERE id=$1 AND status IN ('pending','forwarded')."""

**3-way contract switch (the load-bearing core of this plan):**

```python
async def _dispatch_http_localhost(
    http_client: httpx.AsyncClient,
    row: asyncpg.Record,         # from fetch_pending_for_dispatch
    inapp: InappChannelConfig,   # from InappRecipeIndex
    container_ip: str,
) -> str:
    url = f"http://{container_ip}:{inapp.port}{inapp.endpoint}"
    headers = {"Content-Type": "application/json"}

    match inapp.contract:
        case "openai_compat":
            body = {
                "model": inapp.contract_model_name or "agent",
                "messages": [{"role": "user", "content": row["content"]}],
            }
            if inapp.auth_mode == "bearer" and row["inapp_auth_token"]:
                headers["Authorization"] = f"Bearer {row['inapp_auth_token']}"
            resp = await http_client.post(url, json=body, headers=headers, timeout=600.0)
            resp.raise_for_status()
            data = resp.json()
            try:
                return data["choices"][0]["message"]["content"]
            except (KeyError, IndexError, TypeError) as e:
                raise RuntimeError(f"openai_compat_parse_error:{e}") from e

        case "a2a_jsonrpc":
            body = {
                "jsonrpc": "2.0",
                "id": str(row["id"]),
                "method": "message/send",
                "params": {
                    "message": {
                        "role": "user",
                        "parts": [{"kind": "text", "text": row["content"]}],
                        "messageId": str(row["id"]),
                    }
                },
            }
            resp = await http_client.post(url, json=body, headers=headers, timeout=600.0)
            resp.raise_for_status()
            data = resp.json()
            if "error" in data:
                raise RuntimeError(f"a2a_error:{data['error'].get('message','unknown')}")
            try:
                return data["result"]["artifacts"][0]["parts"][0]["text"]
            except (KeyError, IndexError, TypeError) as e:
                raise RuntimeError(f"a2a_parse_error:{e}") from e

        case "zeroclaw_native":
            body = {"message": row["content"]}
            if row.get("idempotency_key"):
                headers["X-Idempotency-Key"] = row["idempotency_key"]
            headers["X-Session-Id"] = f"inapp:{row['user_id']}:{row['agent_id']}"
            resp = await http_client.post(url, json=body, headers=headers, timeout=600.0)
            resp.raise_for_status()
            data = resp.json()
            try:
                return data["response"]
            except KeyError as e:
                raise RuntimeError(f"zeroclaw_parse_error:{e}") from e

        case other:
            raise RuntimeError(f"unknown_contract:{other}")
```

**Per-row handler (consumes Plan 04 store; INSERTs agent_events for outbox; NO NOTIFY/LISTEN):**

```python
async def _handle_row(state, row: asyncpg.Record) -> None:
    # Pre-dispatch readiness gate (D-37, D-38)
    if (row["container_status"] != "running"
        or row["ready_at"] is None
        or row["stopped_at"] is not None):
        async with state.db.acquire() as conn:
            await mark_failed(conn, row["id"], "container_not_ready")
            await insert_agent_event(
                conn, agent_container_id=row["container_row_id"],
                kind="inapp_outbound_failed",
                payload={"error_type": "container_not_ready", "message": "...", "retry_count": row["attempts"]},
            )
        return

    # Recipe lookup
    inapp = state.recipe_index.get_inapp_block(row["recipe_name"])
    if inapp is None:
        async with state.db.acquire() as conn:
            await mark_failed(conn, row["id"], "recipe_lacks_inapp_channel")
            await insert_agent_event(...)
        return

    # Container IP discovery (cached)
    container_ip = state.recipe_index.get_container_ip(row["container_id"])

    # Mark forwarded BEFORE the bot call (D-28 persist-before-action)
    async with state.db.acquire() as conn:
        await mark_forwarded(conn, row["id"])

    # The single bot call (no retry — D-40)
    try:
        reply = await _dispatch_http_localhost(
            state.bot_http_client, row, inapp, container_ip,
        )
    except httpx.TimeoutException:
        await _terminal_failure(state, row, "bot_timeout")
        return
    except httpx.HTTPStatusError as e:
        await _terminal_failure(state, row, f"bot_5xx:{e.response.status_code}")
        return
    except RuntimeError as e:                # unknown_contract, parse errors, a2a errors
        await _terminal_failure(state, row, str(e))
        return
    except Exception as e:
        log.exception("dispatcher.unexpected_error", id=str(row["id"]))
        await _terminal_failure(state, row, "internal_error")
        return

    if not reply or not reply.strip():
        await _terminal_failure(state, row, "bot_empty")
        return

    # Success: persist + INSERT outbox event
    async with state.db.acquire() as conn:
        async with conn.transaction():
            await mark_done(conn, row["id"], reply)
            await insert_agent_event(
                conn, agent_container_id=row["container_row_id"],
                kind="inapp_outbound",
                payload={"content": reply, "source": "agent"},
            )
```
