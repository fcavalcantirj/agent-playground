    # ====================================================================
    # Phase 22c.3 — inapp chat channel: Redis + httpx + 3 background tasks
    # ====================================================================

    # 1. Redis client (fail-loud at boot; SSE + outbox depend on it).
    app.state.redis = redis_async.from_url(
        settings.redis_url,
        decode_responses=False,        # we publish bytes; SSE handler decodes
        max_connections=20,
    )
    # Per checker WARNING #12: if PING fails, the redis_async client
    # already has open sockets; close them before propagating the
    # exception so the failed boot doesn't leak file descriptors / TCP
    # connections to the (unreachable) host.
    try:
        await app.state.redis.ping()
    except Exception:
        _log.exception("phase22c3.redis.ping_failed_boot")
        try:
            await app.state.redis.aclose()
        except Exception:
            # best-effort close — original exception is the load-bearing one
            pass
        raise   # boot fails — Redis is required for inapp channel

    # 2. Shared httpx.AsyncClient for bot dispatcher (D-40 600s timeout).
    app.state.bot_http_client = httpx.AsyncClient(
        timeout=httpx.Timeout(600.0, connect=5.0),
        limits=httpx.Limits(max_connections=50, max_keepalive_connections=20),
    )

    # 3. Restart sweep (D-31): re-queue rows stuck in 'forwarded' past 15min.
    # Runs BEFORE dispatcher starts so the dispatcher's first tick has a
    # clean state machine to consume from.
    try:
        from .services.inapp_messages_store import restart_sweep
        async with app.state.db.acquire() as _conn:
            _swept = await restart_sweep(_conn, threshold_minutes=15)
            _log.info(
                "phase22c3.restart_sweep",
                extra={"swept": _swept},
            )
    except Exception:
        _log.exception("phase22c3.restart_sweep_failed_nonfatal")
        # Non-fatal: a stuck-row is worse but not catastrophic; reaper
        # will catch them on its next tick.

    # 4. Background tasks (cancellable via inapp_stop event).
    from .services.inapp_dispatcher import dispatcher_loop
    from .services.inapp_reaper import reaper_loop
    from .services.inapp_outbox import outbox_pump_loop

    app.state.inapp_stop = asyncio.Event()
    app.state.inapp_tasks = [
        asyncio.create_task(
            dispatcher_loop(app.state, app.state.inapp_stop),
            name="inapp_dispatcher",
        ),
        asyncio.create_task(
            reaper_loop(app.state, app.state.inapp_stop),
            name="inapp_reaper",
        ),
        asyncio.create_task(
            outbox_pump_loop(app.state, app.state.inapp_stop),
            name="inapp_outbox",
        ),
    ]
    _log.info(
        "phase22c3.lifespan.inapp_tasks_started",
        extra={"task_names": [t.get_name() for t in app.state.inapp_tasks]},
    )

        # Phase 22c.3 — drain inapp tasks first (they may publish to redis
        # during teardown; flush them before closing the redis client).
        try:
            if getattr(app.state, "inapp_stop", None) is not None:
                app.state.inapp_stop.set()
            inapp_tasks = getattr(app.state, "inapp_tasks", [])
            if inapp_tasks:
                done, pending = await asyncio.wait(inapp_tasks, timeout=5.0)
                for p in pending:
                    p.cancel()
