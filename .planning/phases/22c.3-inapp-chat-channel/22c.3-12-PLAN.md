```yaml
channels:
  # ... existing telegram block (if any) stays as-is ...

  # Phase 22c.3 — in-app chat channel (D-20: additive).
  # Empirically verified 2026-04-29 PM (spikes/recipe-openclaw.md): native
  # POST /v1/chat/completions on gateway port 18789 when openclaw.json sets
  # gateway.http.endpoints.chatCompletions.enabled = true AND auth.token is
  # set AND the bot runs `openclaw gateway run` (NOT --allow-unconfigured,
  # which hides chat HTTP). MSV production pattern (provision_ops.go:380-401).
  inapp:
    transport: http_localhost
    port: 18789
    endpoint: /v1/chat/completions
    contract: openai_compat
    # CRITICAL: openclaw rejects raw upstream model ids; clients must send
    # model="openclaw" or "openclaw/<agentId>". The dispatcher (Plan 22c.3-05)
    # reads this field and populates the JSON body's `model`.
    contract_model_name: "openclaw"
    auth_header: "Authorization: Bearer ${INAPP_AUTH_TOKEN}"
    ready_log_regex: "\\[gateway\\] ready \\(\\d+ plugins"
    response_routing: per_session
    persistent_argv_override:
      entrypoint: sh
      argv:
        - "-c"
        - |
          set -e
          mkdir -p /home/node/.openclaw
          cat > /home/node/.openclaw/openclaw.json <<'EOF'
          {
            "agents": {
              "defaults": {
                "model": { "primary": "openrouter/${MODEL}", "fallbacks": [] },
                "timeoutSeconds": 600,
                "thinkingDefault": "high"
              }
            },
            "gateway": {
              "port": 18789,
              "mode": "local",
              "bind": "lan",
              "http": {
                "endpoints": {
                  "chatCompletions": { "enabled": true }
                }
              },
              "auth": {
                "mode": "token",
                "token": "${INAPP_AUTH_TOKEN}"
              },
              "controlUi": { "dangerouslyAllowHostHeaderOriginFallback": true }
            },
            "channels": {},
            "skills": { "install": { "nodeManager": "npm" }, "entries": {} },
            "update": { "checkOnStart": false }
          }
          EOF
          rm -f /home/node/.openclaw/openclaw.json5
          exec openclaw gateway run --port 18789
    activation_env:
      # Runner injects ${INAPP_AUTH_TOKEN} as a per-session bearer token
      # (mints + persists to agent_containers.inapp_auth_token at start time).
      # Runner also injects ${MODEL} = the agent_instance.model column.
      INAPP_AUTH_TOKEN: "${INAPP_AUTH_TOKEN}"
      MODEL: "${MODEL}"
      OPENROUTER_API_KEY: "${OPENROUTER_API_KEY}"
    verified_cells:

          binds POST /v1/chat/completions on port 18789. 401 without bearer,
```
