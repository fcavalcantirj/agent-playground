users(
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT NULL,
  display_name TEXT NOT NULL,
  provider TEXT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
)

agent_instances(
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id),
  recipe_name TEXT NOT NULL,
  model TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  last_run_at TIMESTAMPTZ NULL,
  total_runs INTEGER NOT NULL DEFAULT 0,
  UNIQUE (user_id, recipe_name, model)   -- name 'uq_agent_instances_user_recipe_model'

runs(
  id TEXT PRIMARY KEY,                    -- ULID 26-char
  agent_instance_id UUID NOT NULL REFERENCES agent_instances(id),
  prompt TEXT NOT NULL,
  verdict TEXT NULL,
  category TEXT NULL,
  detail TEXT NULL,
  exit_code INTEGER NULL,
  wall_time_s NUMERIC NULL,
  filtered_payload TEXT NULL,
  stderr_tail TEXT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  completed_at TIMESTAMPTZ NULL
)
CREATE INDEX idx_runs_agent_instance ON runs(agent_instance_id);

idempotency_keys(
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id),
  key TEXT NOT NULL,
  run_id TEXT NOT NULL REFERENCES runs(id),
  verdict_json JSONB NOT NULL,
  request_body_hash TEXT NOT NULL,       -- Pitfall 6 — SHA-256 hex of canonical body
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

CREATE INDEX idx_idempotency_expires ON idempotency_keys(expires_at);

rate_limit_counters(

  count INTEGER NOT NULL DEFAULT 0,

    ```python
    import asyncio
    import os
    from logging.config import fileConfig

    from alembic import context
    from sqlalchemy import MetaData, pool
    from sqlalchemy.ext.asyncio import async_engine_from_config

    config = context.config

    if config.config_file_name is not None:
        fileConfig(config.config_file_name)

    # Normalize DSN: accept postgres://, postgresql://, or postgresql+asyncpg://
    raw_url = os.environ.get("DATABASE_URL")
    if not raw_url:
        raise RuntimeError("DATABASE_URL is required for migrations")
    if raw_url.startswith("postgres://"):
        raw_url = "postgresql+asyncpg://" + raw_url[len("postgres://"):]
    elif raw_url.startswith("postgresql://") and "+asyncpg" not in raw_url:
        raw_url = "postgresql+asyncpg://" + raw_url[len("postgresql://"):]
    config.set_main_option("sqlalchemy.url", raw_url)

    target_metadata = MetaData()  # bare metadata — DDL is authored via op.create_table

    def do_run_migrations(connection):
        context.configure(connection=connection, target_metadata=target_metadata)
        with context.begin_transaction():
            context.run_migrations()

    async def run_async_migrations():
        connectable = async_engine_from_config(
            config.get_section(config.config_ini_section),
            prefix="sqlalchemy.",
            poolclass=pool.NullPool,
        )
        async with connectable.connect() as connection:
            await connection.run_sync(do_run_migrations)
        await connectable.dispose()

    if context.is_offline_mode():
        raise RuntimeError("offline migrations not supported")
    asyncio.run(run_async_migrations())
    ```

            text=True,
            check=True,

        raw = container.get_connection_url()
        dsn = raw.replace("postgresql+psycopg2://", "postgres://").replace("+psycopg2", "")
