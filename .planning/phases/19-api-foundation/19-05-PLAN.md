    ```python
    """Postgres-backed rate limiter — fixed window via advisory lock + upsert counter.

    RESEARCH.md Pattern 4. Accepts 2x burst at window boundaries (documented tradeoff).
    """
    from __future__ import annotations

    import asyncpg


    def _lock_key(subject: str, bucket: str) -> int:
        # Positive int64 — PostgreSQL pg_advisory_xact_lock accepts signed bigint
        combined = f"{subject}|{bucket}"
        h = hash(combined)
        return h & ((1 << 63) - 1)


    async def check_and_increment(conn: asyncpg.Connection,
                                   subject: str,
                                   bucket: str,
                                   limit: int,
                                   window_s: int) -> tuple[bool, int]:
        """Return (allowed, retry_after_s). retry_after_s is 0 when allowed."""
        lock_key = _lock_key(subject, bucket)
        async with conn.transaction():
            await conn.execute("SELECT pg_advisory_xact_lock($1)", lock_key)
            row = await conn.fetchrow(
                """
                INSERT INTO rate_limit_counters (subject, bucket, window_start, count)
                VALUES (
                    $1, $2,
                    date_trunc('second', NOW())
                        - (EXTRACT(EPOCH FROM NOW())::bigint % $3) * INTERVAL '1 second',
                    1
                )
                ON CONFLICT (subject, bucket, window_start)
                DO UPDATE SET count = rate_limit_counters.count + 1
                RETURNING count, window_start,
                          EXTRACT(EPOCH FROM (NOW() - window_start))::int AS age_s
                """,
                subject, bucket, window_s,
            )
            count = row["count"]
            age_s = row["age_s"]
            if count > limit:
                retry_after = max(1, window_s - age_s)
                return (False, retry_after)
            return (True, 0)
    ```

    ```python
    """Postgres-backed Stripe-style idempotency.

    RESEARCH.md Pattern 3. Serializes concurrent duplicates via pg_advisory_xact_lock.
    request_body_hash (Pitfall 6) rejects key reuse with a different body.
    """
    from __future__ import annotations

    import hashlib
    import json
    from typing import Literal, Tuple
    from uuid import UUID

    import asyncpg


    def hash_body(raw_bytes: bytes) -> str:
        return hashlib.sha256(raw_bytes).hexdigest()


    def _lock_key(user_id: UUID, key: str) -> int:
        digest = hashlib.sha256(f"{user_id}:{key}".encode()).digest()[:8]
        return int.from_bytes(digest, "big", signed=True)


    CheckResult = Tuple[Literal["hit", "miss", "mismatch"], dict | None]


    async def check_or_reserve(conn: asyncpg.Connection,
                                user_id: UUID,
                                key: str,
                                body_hash: str) -> CheckResult:
        lock_key = _lock_key(user_id, key)
        async with conn.transaction():
            await conn.execute("SELECT pg_advisory_xact_lock($1)", lock_key)
            row = await conn.fetchrow(
                """
                SELECT run_id, verdict_json, request_body_hash
                  FROM idempotency_keys
                 WHERE user_id = $1 AND key = $2 AND expires_at > NOW()
                """,
                user_id, key,
            )
            if row is None:
                return ("miss", None)
            if row["request_body_hash"] != body_hash:
                return ("mismatch", None)
            cached = row["verdict_json"]
            if isinstance(cached, (str, bytes)):
                cached = json.loads(cached)
            return ("hit", cached)


    async def write_idempotency(conn: asyncpg.Connection,
                                 user_id: UUID,
                                 key: str,
                                 body_hash: str,
                                 run_id: str,
                                 verdict_json: dict,
                                 ttl_hours: int = 24) -> None:
        await conn.execute(
            """
            INSERT INTO idempotency_keys
                (id, user_id, key, run_id, verdict_json,
                 request_body_hash, expires_at)
            VALUES
                (gen_random_uuid(), $1, $2, $3, $4::jsonb, $5,
                 NOW() + ($6 || ' hours')::interval)
            ON CONFLICT (user_id, key) DO NOTHING
            """,
            user_id, key, run_id, json.dumps(verdict_json),
            body_hash, str(ttl_hours),
        )
    ```

    ```python
    D-05 limits: 10/min POST /v1/runs, 120/min POST /v1/lint, 300/min GET /v1/*.
    """

    import json
    import logging

    from starlette.types import ASGIApp, Message, Receive, Scope, Send

    from ..models.errors import ErrorCode, make_error_envelope
    from ..services.rate_limit import check_and_increment

    _log = logging.getLogger("api_server.rate_limit")

    _LIMITS = {
        "runs": (10, 60),
        "lint": (120, 60),
        "get": (300, 60),
    }


    def _bucket_for(scope: Scope) -> str | None:
        method = scope.get("method", "")
        path = scope.get("path", "")
        if method == "POST" and path == "/v1/runs":
            return "runs"
        if method == "POST" and path == "/v1/lint":
            return "lint"
        if method == "GET" and path.startswith("/v1/"):

                if name == b"x-forwarded-for":
                    first = value.decode(errors="ignore").split(",")[0].strip()
                    if first:

    class RateLimitMiddleware:
        def __init__(self, app: ASGIApp):
            self.app = app

        async def __call__(self, scope: Scope, receive: Receive, send: Send) -> None:
            if scope["type"] != "http":
                await self.app(scope, receive, send)
                return
            bucket = _bucket_for(scope)
            if bucket is None:
                await self.app(scope, receive, send)
                return
            limit, window_s = _LIMITS[bucket]

            app = scope["app"]
            trusted = bool(getattr(app.state.settings, "trusted_proxy", False))
            subject = _subject_from_scope(scope, trusted)

            try:
                async with app.state.db.acquire() as conn:
                    allowed, retry_after = await check_and_increment(
                        conn, subject, bucket, limit, window_s,
                    )
            except Exception:
                _log.exception("rate_limit backend error; failing open")
                await self.app(scope, receive, send)
                return

            if allowed:
                await self.app(scope, receive, send)
                return

            body = json.dumps(
                make_error_envelope(
                    ErrorCode.RATE_LIMITED,
                    f"rate limit exceeded for bucket {bucket!r}",
                    param=bucket,
                )
            ).encode()
            await send({
                "type": "http.response.start",
                "status": 429,
                "headers": [
                    (b"content-type", b"application/json"),
                    (b"retry-after", str(retry_after).encode()),
                    (b"content-length", str(len(body)).encode()),
                ],
            })
            await send({"type": "http.response.body", "body": body})
    ```

    ```python
    """Idempotency middleware (Plan 19-05). Replaces Plan 19-02 pass-through stub.

    Applies to POST /v1/runs ONLY. Stripe-style semantics:
    - Same (user_id, key) + same body → replay cached verdict (200)
    - Same (user_id, key) + different body → 422 IDEMPOTENCY_BODY_MISMATCH
    - Different (user_id, key) → miss, run normally, write cache after
    """
    from __future__ import annotations

    import json
    import logging

    from starlette.types import ASGIApp, Message, Receive, Scope, Send

    from ..models.errors import ErrorCode, make_error_envelope
    from ..services.idempotency import check_or_reserve, hash_body, write_idempotency
    from ..services.run_store import ANONYMOUS_USER_ID

    _log = logging.getLogger("api_server.idempotency")


    def _get_header(scope: Scope, name_lower: bytes) -> bytes | None:
        for n, v in scope.get("headers", []):
            if n == name_lower:
                return v
        return None


    async def _read_body(receive: Receive) -> bytes:
        """Drain all http.request messages into a single body buffer."""
        chunks: list[bytes] = []
        while True:
            msg = await receive()
            if msg["type"] != "http.request":
                continue
            if msg.get("body"):
                chunks.append(msg["body"])
            if not msg.get("more_body", False):
                break
        return b"".join(chunks)


    def _replay_receive(body: bytes) -> Receive:
        sent = {"done": False}
        async def _receive():
            if sent["done"]:
                return {"type": "http.disconnect"}
            sent["done"] = True
            return {"type": "http.request", "body": body, "more_body": False}
        return _receive


    async def _send_json(send: Send, status: int, payload: dict) -> None:
        body = json.dumps(payload).encode()
        await send({
            "type": "http.response.start",
            "status": status,
            "headers": [
                (b"content-type", b"application/json"),
                (b"content-length", str(len(body)).encode()),
            ],
        })
        await send({"type": "http.response.body", "body": body})


    class IdempotencyMiddleware:
        def __init__(self, app: ASGIApp):
            self.app = app

        async def __call__(self, scope: Scope, receive: Receive, send: Send) -> None:
            if scope["type"] != "http":
                await self.app(scope, receive, send)
                return
            method = scope.get("method")
            path = scope.get("path")
            if not (method == "POST" and path == "/v1/runs"):
                await self.app(scope, receive, send)
                return

            key_bytes = _get_header(scope, b"idempotency-key")
            if key_bytes is None:
                await self.app(scope, receive, send)
                return
            key = key_bytes.decode(errors="ignore").strip()
            if not key:
                await self.app(scope, receive, send)
                return

            body = await _read_body(receive)
            body_hash = hash_body(body)
            user_id = ANONYMOUS_USER_ID  # phase 19

            app = scope["app"]
            try:
                async with app.state.db.acquire() as conn:
                    tag, cached = await check_or_reserve(conn, user_id, key, body_hash)
            except Exception:
                _log.exception("idempotency backend error; failing open (not cached)")
                await self.app(scope, _replay_receive(body), send)
                return

            if tag == "hit":
                await _send_json(send, 200, cached)
                return
            if tag == "mismatch":
                await _send_json(send, 422, make_error_envelope(
                    ErrorCode.IDEMPOTENCY_BODY_MISMATCH,
                    "Idempotency-Key was used with a different request body",
                    param="Idempotency-Key",
                ))
                return

            # tag == "miss": run normally, then capture response and write_idempotency
            response_status = {"code": 0}
            response_body_chunks: list[bytes] = []

            async def send_wrapper(msg: Message) -> None:
                if msg["type"] == "http.response.start":
                    response_status["code"] = msg["status"]
                elif msg["type"] == "http.response.body" and msg.get("body"):
                    response_body_chunks.append(msg["body"])
                await send(msg)

            await self.app(scope, _replay_receive(body), send_wrapper)

            if response_status["code"] == 200:
                try:
                    resp_body = b"".join(response_body_chunks)
                    verdict_json = json.loads(resp_body)
                    run_id = verdict_json.get("run_id")
                    if run_id:
                        async with app.state.db.acquire() as conn:
                            await write_idempotency(
                                conn, user_id, key, body_hash, run_id, verdict_json,
                            )
                except Exception:
                    _log.exception("idempotency write failed (non-fatal)")
    ```
