    **Test 2 — `$defs` well-formedness + reachability (D-10.2):** this is the RED canary. Structure:

    ```python
    def _collect_refs(node, out):
        """Recursively collect every `$ref` value in the schema tree."""
        if isinstance(node, dict):
            ref = node.get("$ref")
            if isinstance(ref, str):
                out.add(ref)
            for v in node.values():
                _collect_refs(v, out)
        elif isinstance(node, list):
            for v in node:
                _collect_refs(v, out)


    def test_defs_are_well_formed_and_reachable():
        schema = json.loads(SCHEMA_PATH.read_text())
        assert "$defs" in schema, (
            "Phase 18 D-10.2: schema must declare $defs (see D-01 v0_1 + D-02 category)"
        )
        defs = schema["$defs"]
        assert isinstance(defs, dict) and len(defs) >= 1, (
            f"$defs must be a non-empty object, got {type(defs).__name__} with "
            f"{len(defs) if isinstance(defs, dict) else 'N/A'} entries"
        )
        shape_keywords = {"type", "oneOf", "$ref", "enum", "const", "allOf", "anyOf"}
        for name, subschema in defs.items():
            assert isinstance(subschema, dict), (
                f"$defs/{name} must be a JSON object, got {type(subschema).__name__}"
            )
            declared = shape_keywords.intersection(subschema.keys())
            assert declared, (
                f"$defs/{name} must declare at least one shape keyword "
                f"({sorted(shape_keywords)}); got keys {sorted(subschema.keys())}"
            )
        # Reachability: every $def must be referenced somewhere.
        refs = set()
        _collect_refs(schema, refs)
        reachable = {
            r.split("/")[-1]
            for r in refs
            if r.startswith("#/$defs/")
        }
        unreachable = set(defs.keys()) - reachable
        assert not unreachable, (
            f"Unreachable $defs (not referenced by any $ref): {sorted(unreachable)}"
        )
    ```
